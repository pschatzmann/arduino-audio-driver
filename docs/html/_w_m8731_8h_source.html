131 error_t init(codec_config_t* cfg) {
132 if (initialized) {
133 AD_LOGW("The WM8731 has already been initialized");
134 return RESULT_OK;
135 }
136 initialized = true;
137
138 // reset all registers to their default values
139 reset();
140
141 // power up the needed sections, power down the rest
142 updatePower(cfg->input_device, cfg->output_device);
143
144 // analogue path: select dac output, line input, no bypass, no mic
145 analogue_path = DacSelect;
146 if (cfg->input_device != ADC_INPUT_NONE &&
147 cfg->input_device != ADC_INPUT_LINE1 &&
148 cfg->input_device != ADC_INPUT_ALL) {
149 analogue_path |= InputSelectMic;
150 }
151 writeReg(Reg::AnaloguePath, analogue_path);
152
153 // digital path: enable high pass filter, no de-emphasis, unmuted
154 digital_path = 0;
155 writeReg(Reg::DigitalPath, digital_path);
156
157 // default headphone / line out volume (0dB)
158 setVoiceVolume(70);
159
160 // input volume (0dB)
161 writeReg(Reg::LeftLineIn, 0x17 | LineInBoth);
162 writeReg(Reg::RightLineIn, 0x17 | LineInBoth);
163
164 // configure the digital audio interface and sample rate
165 configI2S(CODEC_MODE_BOTH, &cfg->i2s);
166
167 // activate the digital core
168 setActive(true);
169 return RESULT_OK;
170 }

173 error_t deinit(void) {
174 setActive(false);
175 // power down everything (standby)
176 writeReg(Reg::PowerDown, PowerOff);
177 initialized = false;
178 return RESULT_OK;

187 error_t configI2S(codec_mode_t mode, I2SDefinition* iface) {
188 uint16_t format = 0;
189 switch (iface->fmt) {
190 case I2S_LEFT:
191 format |= FormatLeftJustified;
192 break;
193 case I2S_RIGHT:
194 format |= FormatRightJustified;
195 break;
196 case I2S_DSP:
197 format |= FormatDsp;
198 break;
199 case I2S_NORMAL:
200 default:
201 format |= FormatI2S;
202 break;
203 }
204
205 switch (iface->bits) {
206 case BIT_LENGTH_20BITS:
207 format |= WordLength20Bit;
208 break;
209 case BIT_LENGTH_24BITS:
210 format |= WordLength24Bit;
211 break;
212 case BIT_LENGTH_32BITS:
213 format |= WordLength32Bit;
214 break;
215 case BIT_LENGTH_16BITS:
216 default:
217 format |= WordLength16Bit;
218 break;
219 }
220
221 if (iface->mode == MODE_MASTER) {
222 format |= MasterMode;
223 }
224
225 writeReg(Reg::Format, format);
226 writeReg(Reg::SamplingControl, getSamplingControl(iface->rate));
227 return RESULT_OK;
228 }

244 error_t setVoiceVolume(int volume) {
245 if (volume < 0) volume = 0;
246 if (volume > 100) volume = 100;
247 voice_volume = volume;
248 // register range 0x30 (-73dB) .. 0x7F (+6dB), 0..0x2F is also mute
249 uint16_t reg_value = (volume == 0)
250 ? 0x00
251 : 0x30 + ((volume * (0x7F - 0x30)) / 100);
252 error_t ret =
253 writeReg(Reg::LeftHeadphoneOut, reg_value | HeadphoneBoth);
254 ret |= writeReg(Reg::RightHeadphoneOut, reg_value | HeadphoneBoth);
255 return ret;
256 }

259 error_t getVoiceVolume(int* volume) {
260 *volume = voice_volume;
261 return RESULT_OK;
262 }

265 error_t writeReg(Reg reg_addr, uint16_t value) {
266 uint16_t word =
267 ((static_cast<uint8_t>(reg_addr) & 0x7F) << 9) | (value & 0x1FF);
268 uint8_t data[2] = {(uint8_t)((word >> 8) & 0xFF), (uint8_t)(word & 0xFF)};
269 return i2c_bus_write_bytes(i2c_handle, i2c_addr, &data[0], 1, &data[1], 1);
270 }
