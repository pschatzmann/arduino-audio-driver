335 bool configureClocks(uint32_t mclk, uint32_t sample_rate,
336 uint8_t word_size, bool bclk_master) {
337 size_t count = 0;
338 const RateDivs* table = pllDivTable(count);
339 const RateDivs* entry = nullptr;
340
341 for (size_t i = 0; i < count; i++) {
342 if (table[i].rate == sample_rate && table[i].mclk == mclk) {
343 entry = &table[i];
344 break;
345 }
346 }
347 if (entry == nullptr) return false;
348
349 bool rc = true;
350 uint8_t p = entry->pll_p;
351 uint8_t r = 1;
352 uint8_t j = entry->pll_j;
353 uint16_t d = entry->pll_d;
354
355 /* set the PLL dividers */
356 rc &= writePagedReg(PLL_P_R_ADDR, (uint8_t)((1 << 7) | PLL_P(p) | PLL_R(r)));
357 rc &= writePagedReg(PLL_J_ADDR, j);
358 rc &= writePagedReg(PLL_D_MSB_ADDR, (uint8_t)(d >> 8));
359 rc &= writePagedReg(PLL_D_LSB_ADDR, (uint8_t)(d & 0xFF));
360
361 uint8_t madc = entry->madc;
362 uint8_t nadc = entry->nadc;
363 uint8_t aosr = entry->aosr;
364 uint8_t ndac = entry->ndac;
365 uint8_t mdac = entry->mdac;
366 uint8_t dosr = entry->dosr;
367
368 int bclk_div = 0;
369 if (bclk_master) {
370 bclk_div = (dosr * mdac) / (word_size * 2U); /* stereo */
371 if ((bclk_div * word_size * 2) != (dosr * mdac)) return false;
372 rc &= writePagedReg(BCLK_DIV_ADDR,
373 (uint8_t)(BCLK_DIV_POWER_UP | BCLK_DIV((uint8_t)bclk_div)));
374 }
375
376 /* Set clock gen mux and turn on PLL */
377 rc &= writePagedReg(CLOCK_GEN_MUX_ADDR, CLOCK_GEN_MUX_DEFAULT);
378 rc &= updatePagedReg(PLL_P_R_ADDR, PLL_POWER_UP, PLL_POWER_UP);
379
380 /* set NDAC, then MDAC, followed by OSR */
381 rc &= writePagedReg(NDAC_DIV_ADDR, (uint8_t)(NDAC_DIV(ndac) | NDAC_POWER_UP));
382 rc &= writePagedReg(MDAC_DIV_ADDR, (uint8_t)(MDAC_DIV(mdac) | MDAC_POWER_UP));
383 rc &= writePagedReg(OSR_MSB_ADDR, (uint8_t)((dosr >> 8) & OSR_MSB_MASK));
384 rc &= writePagedReg(OSR_LSB_ADDR, (uint8_t)(dosr & OSR_LSB_MASK));
385
386 /* set NADC, MADC, OSR */
387 rc &= writePagedReg(NADC_DIV_ADDR, (uint8_t)(NADC_DIV(nadc) | NADC_POWER_UP));
388 rc &= writePagedReg(MADC_DIV_ADDR, (uint8_t)(MADC_DIV(madc) | MADC_POWER_UP));
389 rc &= writePagedReg(AOSR_ADDR, aosr);
390
391 if (bclk_master) {
392 rc &= writePagedReg(BCLK_DIV_ADDR,
393 (uint8_t)(BCLK_DIV((uint8_t)bclk_div) | BCLK_DIV_POWER_UP));
394 }
395
396 /* calculate MCLK divider to get ~1MHz and setup the timer clock */
397 uint8_t mclk_div = (uint8_t)((mclk + 999999U) / 1000000U);
398 rc &= writePagedReg(TIMER_MCLK_DIV_ADDR,
399 (uint8_t)(TIMER_MCLK_DIV_EN_EXT | TIMER_MCLK_DIV_VAL(mclk_div)));
400
401 return rc;
402 }

448 bool configureOutput() {
449 bool rc = true;
450 uint8_t val = 0;
451
452 bool hp = output_device == DAC_OUTPUT_LINE1 || output_device == DAC_OUTPUT_ALL;
453 bool spk = output_device == DAC_OUTPUT_LINE2 || output_device == DAC_OUTPUT_ALL;
454
455 /* set common mode voltage to 1.65V (half of AVDD), AVDD is typically 3.3V */
456 rc &= readPagedReg(HEADPHONE_DRV_ADDR, val);
457 val &= (uint8_t)~HEADPHONE_DRV_CM_MASK;
458 val |= HEADPHONE_DRV_CM(AIC3110_CM_VOLTAGE_1P65) | HEADPHONE_DRV_RESERVED;
459 rc &= writePagedReg(HEADPHONE_DRV_ADDR, val);
460
461 /* enable pop removal on power down/up */
462 rc &= readPagedReg(HP_OUT_POP_RM_ADDR, val);
463 rc &= writePagedReg(HP_OUT_POP_RM_ADDR, (uint8_t)(val | HP_OUT_POP_RM_ENABLE));
464
465 /* route DAC output to mixer */
466 rc &= writePagedReg(OUTPUT_ROUTING_ADDR, OUTPUT_ROUTING_MIXER);
467
468 /* enable volume control on Headphone out and Speaker out */
469 rc &= writePagedReg(HPL_ANA_VOL_CTRL_ADDR, HPX_ANA_VOL(HPX_ANA_VOL_DEFAULT));
470 rc &= writePagedReg(HPR_ANA_VOL_CTRL_ADDR, HPX_ANA_VOL(HPX_ANA_VOL_DEFAULT));
471 rc &= writePagedReg(SPL_ANA_VOL_CTRL_ADDR, SPX_ANA_VOL(SPX_ANA_VOL_DEFAULT));
472 rc &= writePagedReg(SPR_ANA_VOL_CTRL_ADDR, SPX_ANA_VOL(SPX_ANA_VOL_DEFAULT));
473
474 /* unmute/mute headphone and speaker drivers depending on output_device */
475 rc &= setOutputMute(!hp, AIC3110Channel::HeadphoneLeft);
476 rc &= setOutputMute(!hp, AIC3110Channel::HeadphoneRight);
477 rc &= setOutputMute(!spk, AIC3110Channel::FrontLeft);
478 rc &= setOutputMute(!spk, AIC3110Channel::FrontRight);
479
480 /* power up headphone drivers */
481 rc &= readPagedReg(HEADPHONE_DRV_ADDR, val);
482 if (hp) {
483 val |= HEADPHONE_DRV_POWERUP | HEADPHONE_DRV_RESERVED;
484 } else {
485 val = (uint8_t)((val & (uint8_t)~HEADPHONE_DRV_POWERUP) | HEADPHONE_DRV_RESERVED);
486 }
487 rc &= writePagedReg(HEADPHONE_DRV_ADDR, val);
488
489 /* power up speaker drivers */
490 rc &= readPagedReg(SPEAKER_DRV_ADDR, val);
491 if (spk) {
492 val |= SPEAKER_DRV_POWERUP | SPEAKER_DRV_RESERVED;
493 } else {
494 val = (uint8_t)((val & (uint8_t)~SPEAKER_DRV_POWERUP) | SPEAKER_DRV_RESERVED);
495 }
496 rc &= writePagedReg(SPEAKER_DRV_ADDR, val);
497
498 return rc;
499 }

523 bool setOutputVolume(int vol, AIC3110Channel channel = AIC3110Channel::All) {
524 static const uint8_t vol_array[] = {107, 108, 110, 113, 116, 120,
525 125, 128, 132, 138, 144};
526 static constexpr size_t vol_array_len = sizeof(vol_array) / sizeof(vol_array[0]);
527
528 if (vol > CODEC_OUTPUT_VOLUME_MAX || vol < CODEC_OUTPUT_VOLUME_MIN) return false;
529
530 /* remove sign */
531 vol = -vol;
532
533 uint8_t vol_val;
534 if (vol > HPX_ANA_VOL_FLOOR) {
535 vol_val = HPX_ANA_VOL_FLOOR;
536 } else if (vol > HPX_ANA_VOL_LOW_THRESH) {
537 size_t vol_index = 0;
538 for (; vol_index < vol_array_len; vol_index++) {
539 if (vol_array[vol_index] >= (uint8_t)vol) break;
540 }
541 vol_val = (uint8_t)(HPX_ANA_VOL_LOW_THRESH + vol_index + 1);
542 } else {
543 vol_val = (uint8_t)vol;
544 }
545
546 bool rc = true;
547 switch (channel) {
548 case AIC3110Channel::HeadphoneLeft:
549 return writePagedReg(HPL_ANA_VOL_CTRL_ADDR, HPX_ANA_VOL(vol_val));
550 case AIC3110Channel::HeadphoneRight:
551 return writePagedReg(HPR_ANA_VOL_CTRL_ADDR, HPX_ANA_VOL(vol_val));
552 case AIC3110Channel::FrontLeft:
553 return writePagedReg(SPL_ANA_VOL_CTRL_ADDR, SPX_ANA_VOL(vol_val));
554 case AIC3110Channel::FrontRight:
555 return writePagedReg(SPR_ANA_VOL_CTRL_ADDR, SPX_ANA_VOL(vol_val));
556 case AIC3110Channel::All:
557 rc &= writePagedReg(HPL_ANA_VOL_CTRL_ADDR, HPX_ANA_VOL(vol_val));
558 rc &= writePagedReg(HPR_ANA_VOL_CTRL_ADDR, HPX_ANA_VOL(vol_val));
559 rc &= writePagedReg(SPL_ANA_VOL_CTRL_ADDR, SPX_ANA_VOL(vol_val));
560 rc &= writePagedReg(SPR_ANA_VOL_CTRL_ADDR, SPX_ANA_VOL(vol_val));
