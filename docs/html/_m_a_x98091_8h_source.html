281 bool begin(MAX98091Protocol protocol = MAX98091Protocol::I2S,
282 uint32_t sample_rate = 44100, uint32_t mclk_freq = 12000000,
283 uint8_t word_size = 16, uint8_t channels = 2,
284 bool playback = true) {
285 bool rc = true;
286
287 rc &= softReset();
288
289 /* Put the audio codec into shutdown mode */
290 rc &= writeReg(REG_DEVICE_SHUTDOWN, 0x00);
291
292 rc &= writeReg(REG_DAC_CONTROL, 0x00);
293
294 rc &= writeReg(REG_TDM_CONTROL, 0x00);
295
296 /* Set DLY = 1 to conform to the I2S standard.
297 * DLY is only effective when TDM = 0 */
298 rc &= writeReg(REG_INTERFACE_FORMAT, DLY_MASK);
299
300 /* Configure system clock */
301 rc &= setSystemClock(mclk_freq);
302
303 rc &= setProtocol(protocol);
304 rc &= setAudioFormat(sample_rate, word_size, channels);
305
306 if (playback) rc &= configureOutput();
307
308 /* Bring the audio codec out of shutdown mode */
309 rc &= writeReg(REG_DEVICE_SHUTDOWN, SHDNN_MASK);
310
311 return rc;
312 }

349 bool setAudioFormat(uint32_t sample_rate, uint8_t word_size = 16,
350 uint8_t channels = 2) {
351 bool rc = true;
352 uint8_t sample_rate_mask;
353
354 switch (sample_rate) {
355 case 8000:
356 sample_rate_mask = SR_8K_MASK;
357 break;
358 case 16000:
359 sample_rate_mask = SR_16K_MASK;
360 break;
361 case 32000:
362 sample_rate_mask = SR_32K_MASK;
363 break;
364 case 44100:
365 sample_rate_mask = SR_44K1_MASK;
366 break;
367 case 48000:
368 sample_rate_mask = SR_48K_MASK;
369 break;
370 case 96000:
371 sample_rate_mask = SR_96K_MASK;
372 break;
373 default:
374 return false;
375 }
376 rc &= writeReg(REG_QUICK_SAMPLE_RATE, sample_rate_mask);
377
378 uint8_t mono = (channels == 1) ? 1 : 0;
379 rc &= updateReg(REG_IO_CONFIGURATION, DMONO_MASK, mono);
380
381 // only 16 bit word size is supported by the codec's quick config
382 (void)word_size;
383 rc &= updateReg(REG_INTERFACE_FORMAT, WS_MASK, WS_16B);
384
385 return rc;
386 }

394 bool setSystemClock(uint32_t mclk_freq) {
395 uint8_t psclk;
396 if (mclk_freq >= 10000000 && mclk_freq <= 20000000) {
397 psclk = PSCLK_DIV1;
398 } else if (mclk_freq > 20000000 && mclk_freq <= 40000000) {
399 psclk = PSCLK_DIV2;
400 } else if (mclk_freq > 40000000 && mclk_freq <= 60000000) {
401 psclk = PSCLK_DIV4;
402 } else {
403 return false;
404 }
405
406 bool rc = writeReg(REG_SYSTEM_CLOCK, psclk);
407 rc &= updateReg(REG_MASTER_MODE, MAS_MASK, 0);
408 return rc;
409 }

445 bool configureOutput() {
446 bool rc = true;
447
448 bool hp = output_device == DAC_OUTPUT_LINE1 || output_device == DAC_OUTPUT_ALL;
449 bool spk = output_device == DAC_OUTPUT_LINE2 || output_device == DAC_OUTPUT_ALL;
450
451 rc &= updateReg(REG_IO_CONFIGURATION, SDIEN_MASK, SDIEN_MASK);
452
453 rc &= writeReg(REG_LEFT_SPK_MIXER, MIXSPL_DACL_MASK);
454 rc &= writeReg(REG_RIGHT_SPK_MIXER, MIXSPR_DACR_MASK);
455
456 /* select DAC only source */
457 rc &= writeReg(REG_HP_CONTROL, 0x00);
458
459 /* DAREN, DALEN always on; HPREN/HPLEN and SPREN/SPLEN depend on output_device */
460 uint8_t output_enable = DAREN_MASK | DALEN_MASK;
461 if (hp) output_enable |= HPREN_MASK | HPLEN_MASK;
462 if (spk) output_enable |= SPREN_MASK | SPLEN_MASK;
463 rc &= writeReg(REG_OUTPUT_ENABLE, output_enable);
464
465 rc &= setOutputVolume(DEFAULT_VOLUME, MAX98091Channel::All);
466 rc &= setOutputMute(!hp, MAX98091Channel::HeadphoneLeft);
467 rc &= setOutputMute(!hp, MAX98091Channel::HeadphoneRight);
468 rc &= setOutputMute(!spk, MAX98091Channel::FrontLeft);
469 rc &= setOutputMute(!spk, MAX98091Channel::FrontRight);
470
471 return rc;
472 }

482 bool setVolumeOrMute(MAX98091Channel channel, uint8_t value, bool is_volume) {
483 uint8_t hp_mask = is_volume ? HPVOLL_MASK : HPLM_MASK;
484 uint8_t spk_mask = is_volume ? SPVOLL_MASK : SPLM_MASK;
485
486 switch (channel) {
487 case MAX98091Channel::FrontLeft:
488 return updateReg(REG_LEFT_SPK_VOLUME, spk_mask, value);
489 case MAX98091Channel::FrontRight:
490 return updateReg(REG_RIGHT_SPK_VOLUME, spk_mask, value);
491 case MAX98091Channel::HeadphoneLeft:
492 return updateReg(REG_LEFT_HP_VOLUME, hp_mask, value);
493 case MAX98091Channel::HeadphoneRight:
494 return updateReg(REG_RIGHT_HP_VOLUME, hp_mask, value);
495 case MAX98091Channel::All: {
496 bool rc = true;
497 rc &= updateReg(REG_LEFT_SPK_VOLUME, spk_mask, value);
498 rc &= updateReg(REG_RIGHT_SPK_VOLUME, spk_mask, value);
499 rc &= updateReg(REG_LEFT_HP_VOLUME, hp_mask, value);
500 rc &= updateReg(REG_RIGHT_HP_VOLUME, hp_mask, value);
501 return rc;
502 }
503 default:
504 return false;
505 }
