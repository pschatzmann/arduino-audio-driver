86inline uint8_t WM8978::Write_Reg(uint8_t reg, uint16_t val) {
87 uint8_t byte_val = val;
88 i2c_bus_write_bytes(p_wire, address, &reg, 1, &byte_val, 1);
89 REGVAL_TBL[reg] = val; // Save register value to local

93// WM8978 init
94// Return value: 0, initialization is normal; Other, error code
95inline uint8_t WM8978::Init(void) {
96 uint8_t res;
97 res = Write_Reg(0, 0); // Soft reset WM8978
98 if (res) return 1; // Failed to send command, WM8978 exception
99 // The following are common settings
100 Write_Reg(1, 0X1B); // R1, MICEN is set to 1 (MIC enabled), BIASEN is set to
101 // 1 (simulator works), VMIDSEL[1:0] is set to: 11 (5K)
102 Write_Reg(2, 0X1B0); // R2, ROUT1, LOUT1 output enable (headphones can work),
103 // BOOSTENR, BOOSTENL enable
104 Write_Reg(3, 0X6C); // R3, LOUT2, ROUT2 output enable (speaker operation),
105 // RMIX, LMIX enable
106 Write_Reg(6, 0); // R6, MCLK is provided externally
107 Write_Reg(43, 1 << 4); // R43, INVROUT2 reverse, drives the speaker
108 Write_Reg(
109 47,
110 1 << 8); // R47 setting, PGABOOSTL, left channel MIC gets 20 times gain
111 Write_Reg(
112 48,
113 1 << 8); // R48 setting, PGABOOSTR, right channel MIC gets 20 times gain
114 Write_Reg(49, 1 << 1); // R49, TSDEN, turn on overheating protection
115 Write_Reg(10, 1 << 3); // R10, SOFTMUTE off, 128x sampling, best SNR
116 Write_Reg(14, 1 << 3 | 1 << 8); // R14,ADC 128x sampling rate and enable high
117 // pass filter (3.7Hz cut-off)
118 return 0;
119}

127// WM8978 DAC/ADC configuration
128// adcen: adc enable(1)/disable(0)
129// dacen: dac enable(1)/disable(0)
130inline void WM8978::cfgADDA(uint8_t dacen, uint8_t adcen) {
131 uint16_t regval;
132 regval = WM8978::Read_Reg(3); // Read R3
133 if (dacen)
134 regval |= 3 << 0; // Set the lowest 2 bits of R3 to 1, turn on DACR&DACL
135 else
136 regval &= ~(
137 3 << 0); // Clear the lowest 2 bits of R3 to 0 and turn off DACR&DACL.
138 Write_Reg(3, regval); // Set R3
139 regval = WM8978::Read_Reg(2); // Read R2
140 if (adcen)
141 regval |= 3 << 0; // Set the lowest 2 bits of R2 to 1, turn on ADCR&ADCL
142 else
143 regval &=
144 ~(3 << 0); // Clear the lowest 2 bits of R2 and turn off ADCR&ADCL.
145 Write_Reg(2, regval); // Set R2
146}

148// WM8978 input channel configuration
149// micen: MIC on (1)/off (0)
150// lineinen: Line In on (1)/off (0)
151// auxen: aux on (1)/off (0)
152inline void WM8978::cfgInput(uint8_t micen, uint8_t lineinen, uint8_t auxen) {
153 uint16_t regval;
154 regval = WM8978::Read_Reg(2); // Read R2
155 if (micen)
156 regval |= 3 << 2; // Turn on INPPGAENR, INPPGAENL (MIC's PGA amplification)
157 else
158 regval &= ~(3 << 2); // Close INPPGAENR,INPPGAENL.
159 Write_Reg(2, regval); // Set R2
160
161 regval = WM8978::Read_Reg(44); // Read R44
162 if (micen)
163 regval |= 3 << 4 |
164 3 << 0; // Enable LIN2INPPGA, LIP2INPGA, RIN2INPPGA, RIP2INPGA.
165 else
166 regval &= ~(3 << 4 |
167 3 << 0); // Close LIN2INPPGA, LIP2INPGA, RIN2INPPGA, RIP2INPGA.
168 Write_Reg(44, regval); // Set R44
169
170 if (lineinen)
171 WM8978::setLINEINgain(5); // LINE IN 0dB gain
172 else
173 WM8978::setLINEINgain(0); // Close LINE IN
174 if (auxen)
175 WM8978::setAUXgain(7); // AUX 6dB gain
176 else
177 WM8978::setAUXgain(0); // Close AUX input
178}

180// WM8978 output configuration
181// dacen: DAC output (playback) on (1)/off (0)
182// bpsen: Bypass output (recording, including MIC, LINE IN, AUX, etc.) on
183// (1)/off (0)
184inline void WM8978::cfgOutput(uint8_t dacen, uint8_t bpsen) {
185 uint16_t regval = 0;
186 if (dacen) regval |= 1 << 0; // DAC output enable
187 if (bpsen) {
188 regval |= 1 << 1; // BYPASS enabled
189 regval |= 5 << 2; // 0dB gain
190 }
191 Write_Reg(50, regval); // R50 setting
192 Write_Reg(51, regval); // R51 setting
193}

195// WM8978 MIC gain setting (excluding BOOST's 20dB, MIC-->gain of the ADC input
196// part) gain: 0~63, corresponding to -12dB~35.25dB, 0.75dB/Step
197inline void WM8978::setMICgain(uint8_t gain) {
198 gain &= 0X3F;
199 Write_Reg(45, gain); // R45, left channel PGA setting
200 Write_Reg(46, gain | 1 << 8); // R46, right channel PGA setting

203// WM8978 L2/R2 (that is, Line In) gain setting (L2/R2-->gain of the ADC input
204// part) gain: 0~7, 0 means channel is disabled, 1~7, corresponds to -12dB~6dB,
205// 3dB/Step
206inline void WM8978::setLINEINgain(uint8_t gain) {
207 uint16_t regval;
208 gain &= 0X07;
209 regval = WM8978::Read_Reg(47); // Read R47
210 regval &= ~(7 << 4); // Clear the original settings
211 Write_Reg(47, regval | gain << 4); // Set R47
212 regval = WM8978::Read_Reg(48); // Read R48
213 regval &= ~(7 << 4); // Clear the original settings
214 Write_Reg(48, regval | gain << 4); // Set R48
215}

244inline void WM8978::setHPvol(uint8_t voll, uint8_t volr) {
245 voll &= 0X3F;

247 if (voll == 0) voll |= 1 << 6; // When the volume is 0, mute directly
