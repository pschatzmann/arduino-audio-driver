85 error_t init(codec_config_t* codec_cfg) {
86 GPIO gpio;
87 error_t ret = RESULT_OK;
88 AD_LOGI("Power ON CODEC with GPIO %d", power_pin);
89 gpio.pinMode(power_pin, OUTPUT);
90 gpio.digitalWrite(power_pin, 0);
91 delayMs(20);
92 gpio.digitalWrite(power_pin, 1);
93 delayMs(20);
94
95 ret |= transmitRegisters(
96 tas5805m_registers,
97 sizeof(tas5805m_registers) / sizeof(tas5805m_registers[0]));
98
99 if (ret != 0) { AD_LOGE("Fail to iniitialize tas5805m PA"); return RESULT_FAIL; }
100 return ret;
101 }

110 error_t setVolume(int vol) {
111 int vol_idx = 0;
112
113 if (vol < TAS5805M_VOLUME_MIN) {
114 vol = TAS5805M_VOLUME_MIN;
115 }
116 if (vol > TAS5805M_VOLUME_MAX) {
117 vol = TAS5805M_VOLUME_MAX;
118 }
119 vol_idx = vol / 5;
120
121 uint8_t cmd[2] = {0, 0};
122 error_t ret = RESULT_OK;
123
124 cmd[0] = MASTER_VOL_REG_ADDR;
125 cmd[1] = tas5805m_volume[vol_idx];
126 ret = i2c_bus_write_bytes(i2c_handle, i2c_addr, &cmd[0], 1, &cmd[1], 1);
127 AD_LOGW("volume = 0x%x", cmd[1]);
128 return ret;
129 }

132 error_t getVolume(int* value) {
134 uint8_t cmd[2] = {MASTER_VOL_REG_ADDR, 0x00};
135 error_t ret =
136 i2c_bus_read_bytes(i2c_handle, i2c_addr, &cmd[0], 1, &cmd[1], 1);
137 if (ret != 0) { AD_LOGE("Fail to get volume"); return RESULT_FAIL; }
138 unsigned i;
139 for (i = 0; i < sizeof(tas5805m_volume); i++) {
140 if (cmd[1] >= tas5805m_volume[i]) break;
141 }
142 AD_LOGI("Volume is %d", i * 5);
143 *value = 5 * i;
144 return ret;
145 }

152 error_t setMute(bool enable) {
153 error_t ret = RESULT_OK;
154 uint8_t cmd[2] = {TAS5805M_REG_03, 0x00};
155 ret |= i2c_bus_read_bytes(i2c_handle, i2c_addr, &cmd[0], 1, &cmd[1], 1);
156
157 if (enable) {
158 cmd[1] |= 0x8;
159 } else {
160 cmd[1] &= (~0x08);
161 }
162 ret |= i2c_bus_write_bytes(i2c_handle, i2c_addr, &cmd[0], 1, &cmd[1], 1);
163
164 if (ret != 0) { AD_LOGE("Fail to set mute"); return RESULT_FAIL; }
165 return ret;
166 }

172 error_t setMuteFade(int value) {
173 error_t ret = 0;
174 unsigned char cmd[2] = {MUTE_TIME_REG_ADDR, 0x00};
175 /* Time for register value
176 * 000: 11.5 ms
177 * 001: 53 ms
178 * 010: 106.5 ms
179 * 011: 266.5 ms
180 * 100: 0.535 sec
181 * 101: 1.065 sec
182 * 110: 2.665 sec
183 * 111: 5.33 sec
184 */
185 if (value <= 12) {
186 cmd[1] = 0;
187 } else if (value <= 53) {
188 cmd[1] = 1;
189 } else if (value <= 107) {
190 cmd[1] = 2;
191 } else if (value <= 267) {
192 cmd[1] = 3;
193 } else if (value <= 535) {
194 cmd[1] = 4;
195 } else if (value <= 1065) {
196 cmd[1] = 5;
197 } else if (value <= 2665) {
198 cmd[1] = 6;
199 } else {
200 cmd[1] = 7;
201 }
202 cmd[1] |= (cmd[1] << 4);
203
204 ret |= i2c_bus_write_bytes(i2c_handle, i2c_addr, &cmd[0], 1, &cmd[1], 1);
205 if (ret != 0) { AD_LOGE("Fail to set mute fade"); return RESULT_FAIL; }
206 AD_LOGI("Set mute fade, value:%d, 0x%x", value, cmd[1]);
207 return ret;
208 }

245 error_t transmitRegisters(const tas5805m_cfg_reg_t* conf_buf, int size) {
246 int i = 0;
247 error_t ret = RESULT_OK;
248 while (i < size) {
249 switch (conf_buf[i].offset) {
250 case CFG_META_SWITCH:
251 // Used in legacy applications. Ignored here.
252 break;
253 case CFG_META_DELAY:
254 delayMs(conf_buf[i].value);
255 break;
256 case CFG_META_BURST:
257 ret = i2c_bus_write_bytes(i2c_handle, i2c_addr,
258 (unsigned char*)(&conf_buf[i + 1].offset), 1,
259 (unsigned char*)(&conf_buf[i + 1].value),
260 conf_buf[i].value);
261 i += (conf_buf[i].value / 2) + 1;
262 break;
263 case CFG_END_1:
264 if (CFG_END_2 == conf_buf[i + 1].offset &&
265 CFG_END_3 == conf_buf[i + 2].offset) {
266 AD_LOGI("End of tms5805m reg: %d", i);
267 }
268 break;
269 default:
270 ret = i2c_bus_write_bytes(i2c_handle, i2c_addr,
271 (unsigned char*)(&conf_buf[i].offset), 1,
272 (unsigned char*)(&conf_buf[i].value), 1);
273 break;
274 }
275 i++;
276 }
277 if (ret != RESULT_OK) {
278 AD_LOGE("Fail to load configuration to tas5805m");
279 return RESULT_FAIL;
280 }
281 AD_LOGI("%s: write %d reg done", __FUNCTION__, i);
282 return ret;
283 }
