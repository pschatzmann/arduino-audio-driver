146 error_t setVoiceMute(bool enable) {
147 int regv = readReg(ES8156_DAC_MUTE_REG13);
148 if (enable) {

176 error_t getVoiceVolume(int* volume) {
177 int ret = 0;
178 int regv = 0;
179 *volume = 0;
180 regv = readReg(ES8156_VOLUME_CONTROL_REG14);
181 if (regv == RESULT_FAIL) {
182 ret = RESULT_FAIL;
183 } else {
184 if (regv == dac_vol_handle->reg_value) {
185 *volume = dac_vol_handle->user_volume;
186 } else {
187 *volume = 0;
188 ret = RESULT_FAIL;
189 }
190 }
191 AD_LOGD("Get volume:%.2d reg_value:0x%.2x", *volume, regv);
192 return ret;
193 }

208 error_t standby(void) {
209 error_t ret = 0;
210 ret = writeReg(0x14, 0x00);
211 ret |= writeReg(0x19, 0x02);
212 ret |= writeReg(0x21, 0x1F);
213 ret |= writeReg(0x22, 0x02);
214 ret |= writeReg(0x25, 0x21);
215 ret |= writeReg(0x25, 0xA1);
216 ret |= writeReg(0x18, 0x01);
217 ret |= writeReg(0x09, 0x02);
218 ret |= writeReg(0x09, 0x01);
219 ret |= writeReg(0x08, 0x00);
220 return ret;
221 }

223 error_t resume(void) {
224 error_t ret = 0;
225 ret |= writeReg(0x08, 0x3F);
226 ret |= writeReg(0x09, 0x00);
227 ret |= writeReg(0x18, 0x00);
228
229 ret |= writeReg(0x25, 0x20);
230 ret |= writeReg(0x22, 0x00);
231 ret |= writeReg(0x21, 0x3C);
232 ret |= writeReg(0x19, 0x20);
233 ret |= writeReg(0x14, 179);
234 return ret;
235 }
