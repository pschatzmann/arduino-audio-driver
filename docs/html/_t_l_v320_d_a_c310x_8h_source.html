243 bool configureClocks(uint32_t mclk_freq, uint32_t sample_rate,
244 bool bclk_controller = false, uint8_t word_size = 16) {
245 bool rc = true;
246 uint32_t dac_clk, mod_clk;
247 uint32_t ndac, mdac, bclk_div = 0;
248 int osr, osr_min, osr_max;
249 TLV320DAC310xOsrMultiple osr_multiple = getOsrMultiple(sample_rate);
250
251 if (mclk_freq <= DAC_PROC_CLK_FREQ_MAX) {
252 ndac = 1;
253 } else {
254 ndac = mclk_freq / DAC_PROC_CLK_FREQ_MAX;
255 }
256 dac_clk = mclk_freq / ndac;
257
258 /* find mdac/osr such that mod_clk = sample_rate * osr is an integer
259 * divisor of dac_clk and DAC_MOD_CLK_FREQ_MIN <= mod_clk <=
260 * DAC_MOD_CLK_FREQ_MAX */
261 osr_min = (DAC_MOD_CLK_FREQ_MIN + sample_rate - 1) / sample_rate;
262 osr_max = DAC_MOD_CLK_FREQ_MAX / sample_rate;
263
264 int multiple = (int)osr_multiple;
265 osr_max = (osr_max / multiple) * multiple;
266 osr_min = ((osr_min + multiple - 1) / multiple) * multiple;
267
268 osr = osr_max;
269 mdac = 0;
270 while (osr >= osr_min) {
271 mod_clk = sample_rate * osr;
272 mdac = dac_clk / mod_clk;
273 if ((mdac * mod_clk) == dac_clk) {
274 break;
275 }
276 osr -= multiple;
277 }
278
279 if (osr < osr_min || mdac == 0) {
280 return false;
281 }
282
283 if (bclk_controller) {
284 uint32_t num = (uint32_t)osr * mdac;
285 uint32_t den = (uint32_t)word_size * 2U;
286 bclk_div = num / den;
287 if ((bclk_div * den) != num) {
288 return false;
289 }
290 rc &= writePagedReg(0, BCLK_DIV_ADDR,
291 (uint8_t)(BCLK_DIV_POWER_UP | (bclk_div & BCLK_DIV_MASK)));
292 }
293
294 /* set NDAC, then MDAC, followed by OSR */
295 rc &= writePagedReg(0, NDAC_DIV_ADDR,
296 (uint8_t)((ndac & NDAC_DIV_MASK) | NDAC_POWER_UP_MASK));
297 rc &= writePagedReg(0, MDAC_DIV_ADDR,
298 (uint8_t)((mdac & MDAC_DIV_MASK) | MDAC_POWER_UP_MASK));
299 rc &= writePagedReg(0, OSR_MSB_ADDR, (uint8_t)((osr >> 8) & OSR_MSB_MASK));
300 rc &= writePagedReg(0, OSR_LSB_ADDR, (uint8_t)(osr & OSR_LSB_MASK));
301
302 if (bclk_controller) {
303 rc &= writePagedReg(0, BCLK_DIV_ADDR,
304 (uint8_t)((bclk_div & BCLK_DIV_MASK) | BCLK_DIV_POWER_UP));
305 }
306
307 /* MCLK divider for the internal timer, target ~1MHz */
308 uint32_t mclk_div = (mclk_freq + 999999) / 1000000;
309 rc &= writePagedReg(3, TIMER_MCLK_DIV_ADDR,
310 (uint8_t)(TIMER_MCLK_DIV_EN_EXT | (mclk_div & TIMER_MCLK_DIV_MASK)));
311
312 return rc;
313 }

316 bool configureDai(uint8_t word_size = 16,
317 TLV320DAC310xFormat fmt = TLV320DAC310xFormat::I2S,
318 bool bclk_controller = false, bool wclk_controller = false) {
319 uint8_t val = (uint8_t)(((uint8_t)fmt & IF_CTRL_IFTYPE_MASK) << 6);
320
321 if (bclk_controller) val |= IF_CTRL_BCLK_OUT;
322 if (wclk_controller) val |= IF_CTRL_WCLK_OUT;
323
324 TLV320DAC310xWordLength wlen;
325 switch (word_size) {
326 case 16:
327 wlen = TLV320DAC310xWordLength::Bits16;
328 break;
329 case 20:
330 wlen = TLV320DAC310xWordLength::Bits20;
331 break;
332 case 24:
333 wlen = TLV320DAC310xWordLength::Bits24;
334 break;
335 case 32:
336 wlen = TLV320DAC310xWordLength::Bits32;
337 break;
338 default:
339 return false;
340 }
341 val |= (uint8_t)(((uint8_t)wlen & IF_CTRL_WLEN_MASK) << 4);
342
343 return writePagedReg(0, IF_CTRL1_ADDR, val);
344 }

370 bool configureOutput(TLV320DAC310xCmVoltage cm = TLV320DAC310xCmVoltage::V1P65) {
371 bool rc = true;
372 uint8_t val = 0;
373
374 bool hp = output_device == DAC_OUTPUT_LINE1 || output_device == DAC_OUTPUT_ALL;
375 bool spk = output_device == DAC_OUTPUT_LINE2 || output_device == DAC_OUTPUT_ALL;
376
377 if (!hp && !spk) {
378 /* unroute the DACs and power down both the headphone and speaker drivers */
379 rc &= writePagedReg(1, OUTPUT_ROUTING_ADDR, 0);
380 rc &= readPagedReg(1, HEADPHONE_DRV_ADDR, val);
381 rc &= writePagedReg(1, HEADPHONE_DRV_ADDR,
382 (uint8_t)((val & (uint8_t)~HEADPHONE_DRV_POWERUP) |
383 HEADPHONE_DRV_RESERVED));
384 rc &= readPagedReg(1, SPEAKER_AMP_ADDR, val);
385 rc &= writePagedReg(1, SPEAKER_AMP_ADDR,
386 (uint8_t)((val & (uint8_t)~SPEAKER_AMP_POWERUP) |
387 SPEAKER_AMP_RESERVED));
388 return rc;
389 }
390
391 /* set common mode voltage of the headphone driver (e.g. 1.65V = half of
392 * typical 3.3V AVDD) */
393 rc &= readPagedReg(1, HEADPHONE_DRV_ADDR, val);
394 val &= (uint8_t)~HEADPHONE_DRV_CM_MASK;
395 val |= (uint8_t)((((uint8_t)cm) << 3) & HEADPHONE_DRV_CM_MASK) | HEADPHONE_DRV_RESERVED;
396 rc &= writePagedReg(1, HEADPHONE_DRV_ADDR, val);
397
398 /* enable pop removal on power down/up */
399 rc &= readPagedReg(1, HP_OUT_POP_RM_ADDR, val);
400 rc &= writePagedReg(1, HP_OUT_POP_RM_ADDR, (uint8_t)(val | HP_OUT_POP_RM_ENABLE));
401
402 /* route DAC_L/DAC_R to the mixer amplifiers so that the analog volume
403 * controls have effect; DAC_L feeds both HPL and SPK (SPK is mono),
404 * DAC_R only feeds HPR */
405 val = 0;
406 if (hp || spk) val |= OUTPUT_ROUTING_MIXERL;
407 if (hp) val |= OUTPUT_ROUTING_MIXERR;
408 rc &= writePagedReg(1, OUTPUT_ROUTING_ADDR, val);
409
410 if (hp) {
411 /* enable volume control on headphone out */
412 rc &= writePagedReg(1, HPL_ANA_VOL_CTRL_ADDR, analogVolReg(HPX_ANA_VOL_DEFAULT));
413 rc &= writePagedReg(1, HPR_ANA_VOL_CTRL_ADDR, analogVolReg(HPX_ANA_VOL_DEFAULT));
414
415 /* set headphone driver impedance mode */
416 TLV320DAC310xOutputMode mode = resolveOutputMode();
417 uint8_t drv_ctrl = (mode == TLV320DAC310xOutputMode::Headphone)
418 ? 0x00
419 : HEADPHONE_DRV_LINEOUT;
420 rc &= writePagedReg(1, HEADPHONE_DRV_CTRL_ADDR, drv_ctrl);
421
422 /* unmute headphone drivers */
423 rc &= writePagedReg(1, HPL_DRV_GAIN_CTRL_ADDR, HPX_DRV_UNMUTE);
424 rc &= writePagedReg(1, HPR_DRV_GAIN_CTRL_ADDR, HPX_DRV_UNMUTE);
425
426 /* power up headphone drivers */
427 rc &= readPagedReg(1, HEADPHONE_DRV_ADDR, val);
428 rc &= writePagedReg(1, HEADPHONE_DRV_ADDR,
429 (uint8_t)(val | HEADPHONE_DRV_POWERUP | HEADPHONE_DRV_RESERVED));
430 } else {
431 /* power down headphone drivers */
432 rc &= readPagedReg(1, HEADPHONE_DRV_ADDR, val);
433 rc &= writePagedReg(1, HEADPHONE_DRV_ADDR,
434 (uint8_t)((val & (uint8_t)~HEADPHONE_DRV_POWERUP) |
435 HEADPHONE_DRV_RESERVED));
436 }
437
438 if (spk) {
439 /* route the left analog volume control to the Class-D speaker driver */
440 rc &= writePagedReg(1, SPL_ANA_VOL_CTRL_ADDR, analogVolReg(HPX_ANA_VOL_DEFAULT));
441
442 /* unmute the Class-D speaker driver (default gain = 6dB) */
443 rc &= writePagedReg(1, SPK_DRV_GAIN_CTRL_ADDR, SPK_DRV_UNMUTE);
444
445 /* power up the Class-D speaker driver */
446 rc &= readPagedReg(1, SPEAKER_AMP_ADDR, val);
447 rc &= writePagedReg(1, SPEAKER_AMP_ADDR,
448 (uint8_t)(val | SPEAKER_AMP_POWERUP | SPEAKER_AMP_RESERVED));
449 } else {
450 /* power down the Class-D speaker driver */
451 rc &= readPagedReg(1, SPEAKER_AMP_ADDR, val);
452 rc &= writePagedReg(1, SPEAKER_AMP_ADDR,
453 (uint8_t)((val & (uint8_t)~SPEAKER_AMP_POWERUP) |
454 SPEAKER_AMP_RESERVED));
455 }
456
457 return rc;
458 }

491 bool setOutputVolume(int vol) {
492 static const uint8_t vol_array[] = {107, 108, 110, 113, 116, 120,
493 125, 128, 132, 138, 144};
494 if (vol > OUTPUT_VOLUME_MAX || vol < OUTPUT_VOLUME_MIN) return false;
495
496 /* remove sign: 0 = 0dB .. 156 = -78dB */
497 int v = -vol;
498 uint8_t vol_val;
499 if (v > HPX_ANA_VOL_FLOOR) {
500 vol_val = HPX_ANA_VOL_FLOOR;
501 } else if (v > HPX_ANA_VOL_LOW_THRESH) {
502 size_t idx = 0;
503 size_t n = sizeof(vol_array) / sizeof(vol_array[0]);
504 while (idx < n && vol_array[idx] < (uint8_t)v) idx++;
505 vol_val = (uint8_t)(HPX_ANA_VOL_LOW_THRESH + idx + 1);
506 } else {
507 vol_val = (uint8_t)v;
508 }
509
510 bool rc = true;
511 rc &= writePagedReg(1, HPL_ANA_VOL_CTRL_ADDR, analogVolReg(vol_val));
512 rc &= writePagedReg(1, HPR_ANA_VOL_CTRL_ADDR, analogVolReg(vol_val));
513 return rc;
514 }

536 bool enableBeep(uint32_t freq_hz, uint32_t duration_ms = 500,
537 int8_t left_db = 0, int8_t right_db = 0) {
538 bool rc = true;
539 static constexpr float kTwoPi = 6.28318530717959f;
540
541 float angle = kTwoPi * (float)freq_hz / (float)current_sample_rate;
542 int16_t sin_q15 = (int16_t)(sinf(angle) * 32767.0f);
543 int16_t cos_q15 = (int16_t)(cosf(angle) * 32767.0f);
544
545 uint32_t num_samples = (duration_ms == 0)
546 ? 0xFFFFFFu
547 : (uint32_t)((uint64_t)current_sample_rate * duration_ms / 1000);
548 if (num_samples > 0xFFFFFFu) num_samples = 0xFFFFFFu;
549
550 rc &= writePagedReg(0, BEEP_SIN_MSB_ADDR, (uint8_t)(sin_q15 >> 8));
551 rc &= writePagedReg(0, BEEP_SIN_LSB_ADDR, (uint8_t)(sin_q15 & 0xFF));
552 rc &= writePagedReg(0, BEEP_COS_MSB_ADDR, (uint8_t)(cos_q15 >> 8));
553 rc &= writePagedReg(0, BEEP_COS_LSB_ADDR, (uint8_t)(cos_q15 & 0xFF));
554
555 rc &= writePagedReg(0, BEEP_LEN_MSB_ADDR, (uint8_t)((num_samples >> 16) & 0xFF));
556 rc &= writePagedReg(0, BEEP_LEN_MID_ADDR, (uint8_t)((num_samples >> 8) & 0xFF));
557 rc &= writePagedReg(0, BEEP_LEN_LSB_ADDR, (uint8_t)(num_samples & 0xFF));
558
559 rc &= writePagedReg(0, R_BEEP_GEN_ADDR, beepVolReg(right_db));
560 /* write L register last — the enable bit starts the tone */
561 rc &= writePagedReg(0, L_BEEP_GEN_ADDR,
562 (uint8_t)(BEEP_GEN_EN_BEEP | beepVolReg(left_db)));
563 return rc;
564 }

655 }

660 if (sample_rate >= 96000) return TLV320DAC310xOsrMultiple::Multiple4;
661 return TLV320DAC310xOsrMultiple::Multiple8;
662 }
663
665 bool selectPage(uint8_t page) {
666 if (active_page == page) return true;
667 if (!writeReg(PAGE_CONTROL_ADDR, page)) return false;
668 active_page = page;
669 return true;
670 }

680 if (!selectPage(page)) return false;
681 return readReg(reg, value);
682 }

686 if (!selectPage(page)) return false;
687 return updateReg(reg, mask, value);
