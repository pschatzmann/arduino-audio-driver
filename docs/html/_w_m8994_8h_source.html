115 uint32_t init(uint16_t OutputInputDevice, uint8_t Volume,
116 uint32_t AudioFreq) {
117 uint32_t counter = 0;
118 uint16_t output_device = OutputInputDevice & 0xFF;
119 uint16_t input_device = OutputInputDevice & 0xFF00;
120 uint16_t power_mgnt_reg_1 = 0;
121
122 /* wm8994 Errata Work-Arounds */
123 counter += writeReg16(0x102, 0x0003);
124 counter += writeReg16(0x817, 0x0000);
125 counter += writeReg16(0x102, 0x0000);
126
127 /* Enable VMID soft start (fast), Start-up Bias Current Enabled */
128 counter += writeReg16(0x39, 0x006C);
129
130 /* Enable bias generator, Enable VMID */
131 counter += writeReg16(0x01, 0x0003);
132
133 /* Add Delay */
134 delayMs(50);
135
136 /* Path Configurations for output */
137 if (output_device > 0) {
138 outputEnabled = 1;
139 switch (output_device) {
140 case OUTPUT_DEVICE_SPEAKER:
141 /* Enable DAC1 (Left), Enable DAC1 (Right),
142 Disable DAC2 (Left), Disable DAC2 (Right)*/
143 counter += writeReg16(0x05, 0x0C0C);
144
145 /* Enable the AIF1 Timeslot 0 (Left) to DAC 1 (Left) mixer path */
146 counter += writeReg16(0x601, 0x0000);
147
148 /* Enable the AIF1 Timeslot 0 (Right) to DAC 1 (Right) mixer path */
149 counter += writeReg16(0x602, 0x0000);
150
151 /* Disable the AIF1 Timeslot 1 (Left) to DAC 2 (Left) mixer path */
152 counter += writeReg16(0x604, 0x0002);
153
154 /* Disable the AIF1 Timeslot 1 (Right) to DAC 2 (Right) mixer path */
155 counter += writeReg16(0x605, 0x0002);
156 break;
157
158 case OUTPUT_DEVICE_HEADPHONE:
159 /* Disable DAC1 (Left), Disable DAC1 (Right),
160 Enable DAC2 (Left), Enable DAC2 (Right)*/
161 counter += writeReg16(0x05, 0x0303);
162
163 /* Enable the AIF1 Timeslot 0 (Left) to DAC 1 (Left) mixer path */
164 counter += writeReg16(0x601, 0x0001);
165
166 /* Enable the AIF1 Timeslot 0 (Right) to DAC 1 (Right) mixer path */
167 counter += writeReg16(0x602, 0x0001);
168
169 /* Disable the AIF1 Timeslot 1 (Left) to DAC 2 (Left) mixer path */
170 counter += writeReg16(0x604, 0x0000);
171
172 /* Disable the AIF1 Timeslot 1 (Right) to DAC 2 (Right) mixer path */
173 counter += writeReg16(0x605, 0x0000);
174 break;
175
176 case OUTPUT_DEVICE_BOTH:
177 /* Enable DAC1 (Left), Enable DAC1 (Right),
178 also Enable DAC2 (Left), Enable DAC2 (Right)*/
179 counter += writeReg16(0x05, 0x0303 | 0x0C0C);
180
181 /* Enable the AIF1 Timeslot 0 (Left) to DAC 1 (Left) mixer path */
182 counter += writeReg16(0x601, 0x0001);
183
184 /* Enable the AIF1 Timeslot 0 (Right) to DAC 1 (Right) mixer path */
185 counter += writeReg16(0x602, 0x0001);
186
187 /* Enable the AIF1 Timeslot 1 (Left) to DAC 2 (Left) mixer path */
188 counter += writeReg16(0x604, 0x0002);
189
190 /* Enable the AIF1 Timeslot 1 (Right) to DAC 2 (Right) mixer path */
191 counter += writeReg16(0x605, 0x0002);
192 break;
193
194 case OUTPUT_DEVICE_AUTO:
195 default:
196 /* Disable DAC1 (Left), Disable DAC1 (Right),
197 Enable DAC2 (Left), Enable DAC2 (Right)*/
198 counter += writeReg16(0x05, 0x0303);
199
200 /* Enable the AIF1 Timeslot 0 (Left) to DAC 1 (Left) mixer path */
201 counter += writeReg16(0x601, 0x0001);
202
203 /* Enable the AIF1 Timeslot 0 (Right) to DAC 1 (Right) mixer path */
204 counter += writeReg16(0x602, 0x0001);
205
206 /* Disable the AIF1 Timeslot 1 (Left) to DAC 2 (Left) mixer path */
207 counter += writeReg16(0x604, 0x0000);
208
209 /* Disable the AIF1 Timeslot 1 (Right) to DAC 2 (Right) mixer path */
210 counter += writeReg16(0x605, 0x0000);
211 break;
212 }
213 } else {
214 outputEnabled = 0;
215 }
216
217 /* Path Configurations for input */
218 if (input_device > 0) {
219 inputEnabled = 1;
220 switch (input_device) {
221 case INPUT_DEVICE_DIGITAL_MICROPHONE_2:
222 /* Enable AIF1ADC2 (Left), Enable AIF1ADC2 (Right)
223 * Enable DMICDAT2 (Left), Enable DMICDAT2 (Right)
224 * Enable Left ADC, Enable Right ADC */
225 counter += writeReg16(0x04, 0x0C30);
226
227 /* Enable AIF1 DRC2 Signal Detect & DRC in AIF1ADC2 Left/Right Timeslot 1 */
228 counter += writeReg16(0x450, 0x00DB);
229
230 /* Disable IN1L, IN1R, IN2L, IN2R, Enable Thermal sensor & shutdown */
231 counter += writeReg16(0x02, 0x6000);
232
233 /* Enable the DMIC2(Left) to AIF1 Timeslot 1 (Left) mixer path */
234 counter += writeReg16(0x608, 0x0002);
235
236 /* Enable the DMIC2(Right) to AIF1 Timeslot 1 (Right) mixer path */
237 counter += writeReg16(0x609, 0x0002);
238
239 /* GPIO1 pin configuration GP1_DIR = output, GP1_FN = AIF1 DRC2 signal detect */
240 counter += writeReg16(0x700, 0x000E);
241 break;
242
243 case INPUT_DEVICE_INPUT_LINE_1:
244 /* Enable AIF1ADC1 (Left), Enable AIF1ADC1 (Right)
245 * Enable Left ADC, Enable Right ADC */
246 counter += writeReg16(0x04, 0x0303);
247
248 /* Enable AIF1 DRC1 Signal Detect & DRC in AIF1ADC1 Left/Right Timeslot 0 */
249 counter += writeReg16(0x440, 0x00DB);
250
251 /* Enable IN1L and IN1R, Disable IN2L and IN2R, Enable Thermal sensor & shutdown */
252 counter += writeReg16(0x02, 0x6350);
253
254 /* Enable the ADCL(Left) to AIF1 Timeslot 0 (Left) mixer path */
255 counter += writeReg16(0x606, 0x0002);
256
257 /* Enable the ADCR(Right) to AIF1 Timeslot 0 (Right) mixer path */
258 counter += writeReg16(0x607, 0x0002);
259
260 /* GPIO1 pin configuration GP1_DIR = output, GP1_FN = AIF1 DRC1 signal detect */
261 counter += writeReg16(0x700, 0x000D);
262 break;
263
264 case INPUT_DEVICE_DIGITAL_MICROPHONE_1:
265 case INPUT_DEVICE_INPUT_LINE_2:
266 default:
267 /* Actually, no other input devices supported */
268 counter++;
269 break;
270 }
271 } else {
272 inputEnabled = 0;
273 }
274
275 /* Clock Configurations */
276 switch (AudioFreq) {
277 case AUDIO_FREQUENCY_8K:
278 /* AIF1 Sample Rate = 8 (KHz), ratio=256 */
279 counter += writeReg16(0x210, 0x0003);
280 break;
281
282 case AUDIO_FREQUENCY_16K:
283 /* AIF1 Sample Rate = 16 (KHz), ratio=256 */
284 counter += writeReg16(0x210, 0x0033);
285 break;
286
287 case AUDIO_FREQUENCY_48K:
288 /* AIF1 Sample Rate = 48 (KHz), ratio=256 */
289 counter += writeReg16(0x210, 0x0083);
290 break;
291
292 case AUDIO_FREQUENCY_96K:
293 /* AIF1 Sample Rate = 96 (KHz), ratio=256 */
294 counter += writeReg16(0x210, 0x00A3);
295 break;
296
297 case AUDIO_FREQUENCY_11K:
298 /* AIF1 Sample Rate = 11.025 (KHz), ratio=256 */
299 counter += writeReg16(0x210, 0x0013);
300 break;
301
302 case AUDIO_FREQUENCY_22K:
303 /* AIF1 Sample Rate = 22.050 (KHz), ratio=256 */
304 counter += writeReg16(0x210, 0x0043);
305 break;
306
307 case AUDIO_FREQUENCY_44K:
308 /* AIF1 Sample Rate = 44.1 (KHz), ratio=256 */
309 counter += writeReg16(0x210, 0x0073);
310 break;
311
312 default:
313 /* AIF1 Sample Rate = 48 (KHz), ratio=256 */
314 counter += writeReg16(0x210, 0x0083);
315 break;
316 }
317 /* AIF1 Word Length = 16-bits, AIF1 Format = I2S (Default Register Value) */
318 counter += writeReg16(0x300, 0x4010);
319
320 /* slave mode */
321 counter += writeReg16(0x302, 0x0000);
322
323 /* Enable the DSP processing clock for AIF1, Enable the core clock */
324 counter += writeReg16(0x208, 0x000A);
325
326 /* Enable AIF1 Clock, AIF1 Clock Source = MCLK1 pin */
327 counter += writeReg16(0x200, 0x0001);
328
329 if (output_device > 0) /* Audio output selected */
330 {
331 /* Analog Output Configuration */
332
333 /* Enable SPKRVOL PGA, Enable SPKMIXR, Enable SPKLVOL PGA, Enable SPKMIXL */
334 counter += writeReg16(0x03, 0x0300);
335
336 /* Left Speaker Mixer Volume = 0dB */
337 counter += writeReg16(0x22, 0x0000);
338
339 /* Speaker output mode = Class D, Right Speaker Mixer Volume = 0dB ((0x23, 0x0100) = class AB)*/
340 counter += writeReg16(0x23, 0x0000);
341
342 /* Unmute DAC2 (Left) to Left Speaker Mixer (SPKMIXL) path,
343 Unmute DAC2 (Right) to Right Speaker Mixer (SPKMIXR) path */
344 counter += writeReg16(0x36, 0x0300);
345
346 /* Enable bias generator, Enable VMID, Enable SPKOUTL, Enable SPKOUTR */
347 counter += writeReg16(0x01, 0x3003);
348
349 /* Headphone/Speaker Enable */
350
351 /* Enable Class W, Class W Envelope Tracking = AIF1 Timeslot 0 */
352 counter += writeReg16(0x51, 0x0005);
353
354 /* Enable bias generator, Enable VMID, Enable HPOUT1 (Left) and Enable HPOUT1 (Right) input stages */
355 /* idem for Speaker */
356 power_mgnt_reg_1 |= 0x0303 | 0x3003;
357 counter += writeReg16(0x01, power_mgnt_reg_1);
358
359 /* Enable HPOUT1 (Left) and HPOUT1 (Right) intermediate stages */
360 counter += writeReg16(0x60, 0x0022);
361
362 /* Enable Charge Pump */
363 counter += writeReg16(0x4C, 0x9F25);
364
365 /* Add Delay */
366 delayMs(15);
367
368 /* Select DAC1 (Left) to Left Headphone Output PGA (HPOUT1LVOL) path */
369 counter += writeReg16(0x2D, 0x0001);
370
371 /* Select DAC1 (Right) to Right Headphone Output PGA (HPOUT1RVOL) path */
372 counter += writeReg16(0x2E, 0x0001);
373
374 /* Enable Left Output Mixer (MIXOUTL), Enable Right Output Mixer (MIXOUTR) */
375 /* idem for SPKOUTL and SPKOUTR */
376 counter += writeReg16(0x03, 0x0030 | 0x0300);
377
378 /* Enable DC Servo and trigger start-up mode on left and right channels */
379 counter += writeReg16(0x54, 0x0033);
380
381 /* Add Delay */
382 delayMs(250);
383
384 /* Enable HPOUT1 (Left) and HPOUT1 (Right) intermediate and output stages. Remove clamps */
385 counter += writeReg16(0x60, 0x00EE);
386
387 /* Unmutes */
388
389 /* Unmute DAC 1 (Left) */
390 counter += writeReg16(0x610, 0x00C0);
391
392 /* Unmute DAC 1 (Right) */
393 counter += writeReg16(0x611, 0x00C0);
394
395 /* Unmute the AIF1 Timeslot 0 DAC path */
396 counter += writeReg16(0x420, 0x0000);
397
398 /* Unmute DAC 2 (Left) */
399 counter += writeReg16(0x612, 0x00C0);
400
401 /* Unmute DAC 2 (Right) */
402 counter += writeReg16(0x613, 0x00C0);
403
404 /* Unmute the AIF1 Timeslot 1 DAC2 path */
405 counter += writeReg16(0x422, 0x0000);
406
407 /* Volume Control */
408 setVolume(Volume);
409 }
410
411 if (input_device > 0) /* Audio input selected */
412 {
413 if ((input_device == INPUT_DEVICE_DIGITAL_MICROPHONE_1) ||
414 (input_device == INPUT_DEVICE_DIGITAL_MICROPHONE_2)) {
415 /* Enable Microphone bias 1 generator, Enable VMID */
416 power_mgnt_reg_1 |= 0x0013;
417 counter += writeReg16(0x01, power_mgnt_reg_1);
418
419 /* ADC oversample enable */
420 counter += writeReg16(0x620, 0x0002);
421
422 /* AIF ADC2 HPF enable, HPF cut = voice mode 1 fc=127Hz at fs=8kHz */
423 counter += writeReg16(0x411, 0x3800);
424 } else if ((input_device == INPUT_DEVICE_INPUT_LINE_1) ||
425 (input_device == INPUT_DEVICE_INPUT_LINE_2)) {
426 /* Enable normal bias generator, Enable VMID */
427 power_mgnt_reg_1 |= 0x0003;
428 counter += writeReg16(0x01, power_mgnt_reg_1);
429
430 /* Disable mute on IN1L, IN1L Volume = +0dB */
431 counter += writeReg16(0x18, 0x000B);
432
433 /* Disable mute on IN1R, IN1R Volume = +0dB */
434 counter += writeReg16(0x1A, 0x000B);
435
436 /* Disable mute on IN1L_TO_MIXINL, Gain = +0dB */
437 counter += writeReg16(0x29, 0x0025);
438
439 /* Disable mute on IN1R_TO_MIXINL, Gain = +0dB */
440 counter += writeReg16(0x2A, 0x0025);
441
442 /* IN1LN_TO_IN1L, IN1LP_TO_VMID, IN1RN_TO_IN1R, IN1RP_TO_VMID */
443 counter += writeReg16(0x28, 0x0011);
444
445 /* AIF ADC1 HPF enable, HPF cut = hifi mode fc=4Hz at fs=48kHz */
446 counter += writeReg16(0x410, 0x1800);
447 }
448 /* Volume Control */
449 setVolume(Volume);
450 }
451 /* Return communication control value */
452 return counter;
453 }

515 uint32_t stop(uint32_t CodecPdwnMode) {
516 uint32_t counter = 0;
517
518 if (outputEnabled != 0) {
519 /* Mute the output first */
520 counter += setMute(AUDIO_MUTE_ON);
521
522 if (CodecPdwnMode == CODEC_PDWN_SW) {
523 /* Only output mute required*/
524 } else /* CODEC_PDWN_HW */
525 {
526 /* Mute the AIF1 Timeslot 0 DAC1 path */
527 counter += writeReg16(0x420, 0x0200);
528
529 /* Mute the AIF1 Timeslot 1 DAC2 path */
530 counter += writeReg16(0x422, 0x0200);
531
532 /* Disable DAC1L_TO_HPOUT1L */
533 counter += writeReg16(0x2D, 0x0000);
534
535 /* Disable DAC1R_TO_HPOUT1R */
536 counter += writeReg16(0x2E, 0x0000);
537
538 /* Disable DAC1 and DAC2 */
539 counter += writeReg16(0x05, 0x0000);
540

556 uint32_t setVolume(uint8_t Volume) {
557 uint32_t counter = 0;
558 uint8_t convertedvol = wm8994VolumeConvert(Volume);
559
560 /* Output volume */
561 if (outputEnabled != 0) {
562 if (convertedvol > 0x3E) {
563 /* Unmute audio codec */
564 counter += setMute(AUDIO_MUTE_OFF);
565
566 /* Left Headphone Volume */
567 counter += writeReg16(0x1C, 0x3F | 0x140);
568
569 /* Right Headphone Volume */
570 counter += writeReg16(0x1D, 0x3F | 0x140);
571
572 /* Left Speaker Volume */
573 counter += writeReg16(0x26, 0x3F | 0x140);
574
575 /* Right Speaker Volume */
576 counter += writeReg16(0x27, 0x3F | 0x140);
577 } else if (Volume == 0) {
578 /* Mute audio codec */
579 counter += setMute(AUDIO_MUTE_ON);
580 } else {
581 /* Unmute audio codec */
582 counter += setMute(AUDIO_MUTE_OFF);
583
584 /* Left Headphone Volume */
585 counter += writeReg16(0x1C, convertedvol | 0x140);
586
587 /* Right Headphone Volume */
588 counter += writeReg16(0x1D, convertedvol | 0x140);
589
590 /* Left Speaker Volume */
591 counter += writeReg16(0x26, convertedvol | 0x140);
592
593 /* Right Speaker Volume */
594 counter += writeReg16(0x27, convertedvol | 0x140);
595 }
596 }
597
598 /* Input volume */
599 if (inputEnabled != 0) {
600 convertedvol = volumeInConvert(Volume);
601
602 /* Left AIF1 ADC1 volume */
603 counter += writeReg16(0x400, convertedvol | 0x100);
604
605 /* Right AIF1 ADC1 volume */
606 counter += writeReg16(0x401, convertedvol | 0x100);
607
608 /* Left AIF1 ADC2 volume */
609 counter += writeReg16(0x404, convertedvol | 0x100);
610
611 /* Right AIF1 ADC2 volume */
612 counter += writeReg16(0x405, convertedvol | 0x100);
613 }
614 return counter;
