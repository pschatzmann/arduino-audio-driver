602 bool applySampleRateClocks(const SRateAttr* srate, int n1_sel,
603 int mclk_mult_sel, int n2_sel) {
604 if (!srate || n1_sel < 0 || n1_sel >= 3 || n2_sel < 0 || n2_sel >= 5) {
605 AD_LOGI("[NAU8325] Invalid N1/N2 divider index.");
606 return false;
607 }
608
609 AD_LOGI("--- Debugging clock dividers ---");
610 AD_LOGI("n1_sel=%d (val=%d)", n1_sel, mclk_n1_div[n1_sel].val);
611 AD_LOGI("n2_sel=%d (val=%d)", n2_sel, mclk_n2_div[n2_sel].val);
612 AD_LOGI("mult_sel=%d", mclk_mult_sel);
613
614 writeRegisterBits(NAU8325_R40_CLK_DET_CTRL,
615 NAU8325_REG_SRATE_MASK | NAU8325_REG_DIV_MAX,
616 (srate->range << NAU8325_REG_SRATE_SFT) |
617 (srate->max ? NAU8325_REG_DIV_MAX : 0));
618 uint16_t clk_ctrl_val = 0;
619 clk_ctrl_val |= mclk_n2_div[n2_sel].val;
620 clk_ctrl_val |= mclk_n1_div[n1_sel].val << NAU8325_CLK_MUL_SRC_SFT;
621 if (mclk_mult_sel >= 0 && mclk_mult_sel <= 3)
622 clk_ctrl_val |= mclk_n3_mult[mclk_mult_sel].val << NAU8325_MCLK_SEL_SFT;
623
624 clk_ctrl_val |= 0x1000; // DAC_CLK = MCLK/2 (required!)
625
626 writeRegister(NAU8325_R03_CLK_CTRL, clk_ctrl_val);
627 writeRegister(NAU8325_R03_CLK_CTRL, clk_ctrl_val);
628
629 switch (mclk_mult_sel) {
630 case 2:
631 writeRegisterBits(NAU8325_R65_ANALOG_CONTROL_5, NAU8325_MCLK4XEN_EN,
632 NAU8325_MCLK4XEN_EN);
633 break;
634 case 3:
635 writeRegisterBits(NAU8325_R65_ANALOG_CONTROL_5,
636 NAU8325_MCLK4XEN_EN | NAU8325_MCLK8XEN_EN,
637 NAU8325_MCLK4XEN_EN | NAU8325_MCLK8XEN_EN);
638 break;
639 default:
640 writeRegisterBits(NAU8325_R65_ANALOG_CONTROL_5,
641 NAU8325_MCLK4XEN_EN | NAU8325_MCLK8XEN_EN, 0);
642 break;
643 }
644
645 return true;
646 }

649 int getMclkRatioAndN2Index(const SRateAttr* srate, int mclk_hz, int& n2_sel_out) {
650 int ratio = NAU8325_MCLK_FS_RATIO_NUM;
651
652 for (int i = 0; i < 5; i++) {
653 int div = mclk_n2_div[i].param;
654 int mclk_src = mclk_hz >> div;
655
656 if (srate->mclk_src[NAU8325_MCLK_FS_RATIO_256] == (uint32_t)mclk_src) {
657 ratio = NAU8325_MCLK_FS_RATIO_256;
658 n2_sel_out = i;
659 break;
660 }
661 if (srate->mclk_src[NAU8325_MCLK_FS_RATIO_400] == (uint32_t)mclk_src) {
662 ratio = NAU8325_MCLK_FS_RATIO_400;
663 n2_sel_out = i;
664 break;
665 }
666 if (srate->mclk_src[NAU8325_MCLK_FS_RATIO_500] == (uint32_t)mclk_src) {
667 ratio = NAU8325_MCLK_FS_RATIO_500;
668 n2_sel_out = i;
669 break;
670 }
671 }
672
673 return ratio;
674 }

688 bool chooseClockSource(int fs, int mclk, const SRateAttr*& srate,
689 int& n1_sel, int& mult_sel, int& n2_sel) {
690 if (fs <= 0 || mclk <= 0) {
691 AD_LOGI("[NAU8325] Invalid fs or mclk.");
692 return false;
693 }
694
695 srate = getSRateAttr(fs);
696 if (!srate) {
697 AD_LOGI("[NAU8325] Unsupported fs: %d\n", fs);
698 return false;
699 }
700
701 int ratio = getMclkRatioAndN2Index(srate, mclk, n2_sel);
702 if (ratio != NAU8325_MCLK_FS_RATIO_NUM) {
703 n1_sel = 0;
704 mult_sel = -1;
705 AD_LOGI("[NAU8325] Direct match: fs=%d, mclk=%d → N2=%d (Ratio=%d)\n", fs,
706 mclk, n2_sel, ratio);
707 return true;
708 }
709
710 int mclk_max = 0;
711 int best_n1 = -1, best_mult = -1, best_n2 = -1;
712
713 for (int i = 0; i < 3; ++i) {
714 for (int j = 0; j < 4; ++j) {
715 int m = (mclk << mclk_n3_mult[j].param) / mclk_n1_div[i].param;
716 int r = getMclkRatioAndN2Index(srate, m, n2_sel);
717
718 if (r != NAU8325_MCLK_FS_RATIO_NUM && (m > mclk_max || best_n1 < i)) {
719 mclk_max = m;
720 best_n1 = i;
721 best_mult = j;
722 best_n2 = n2_sel;
723 }
724 }
725 }
726
727 if (mclk_max > 0) {
728 n1_sel = best_n1;
729 mult_sel = best_mult;
730 n2_sel = best_n2;
731
732 AD_LOGI(
733 "[NAU8325] Matched via N1/N3 combo: fs=%d, mclk=%d → N1=%d, "
734 "N3(mult)=%d, N2=%d\n",
735 fs, mclk, n1_sel, mult_sel, n2_sel);
736 AD_LOGI("[NAU8325] chooseClockSource(): fs=%d, mclk=%d\n", fs, mclk);
737
738 return true;
739 }
740
741 AD_LOGI("[NAU8325] Failed to match MCLK %d Hz with fs %d Hz\n", mclk, fs);
742 return false;
743 }

746 bool configureClocks(int fs, int mclk) {
747 const SRateAttr* srate = nullptr;
748 int n1_sel = 0, mult_sel = -1, n2_sel = 0;
749
750 if (!chooseClockSource(fs, mclk, srate, n1_sel, mult_sel, n2_sel)) {
751 AD_LOGI("[NAU8325] Failed to choose clock source for fs=%d, mclk=%d\n",

757 AD_LOGI("[NAU8325] Failed to apply sample rate clocks.");
758 return false;
759 }
760 AD_LOGI("[NAU8325] configureClocks(): fs=%u, mclk=%u\n", fs, mclk);
761
762 return true;

782 bool configureAudio(uint32_t fs, uint32_t mclk, uint8_t bits_per_sample) {
783 this->fs = fs;
784 this->mclk = mclk;
785
786 if (!configureClocks(fs, mclk)) {
787 return false;
788 }
789
790 writeRegisterBits(NAU8325_R29_DAC_CTRL1, NAU8325_DAC_OVERSAMPLE_MASK,
791 NAU8325_DAC_OVERSAMPLE_128);
792
793 const OsrAttr* osr = getCurrentOSR();
794 if (!osr || osr->osr == 0 || (fs * osr->osr > CLK_DA_AD_MAX)) {
795 AD_LOGI("[NAU8325] Invalid OSR or fs × OSR exceeds max.");
796 return false;
797 }
798
799 writeRegisterBits(NAU8325_R03_CLK_CTRL, NAU8325_CLK_DAC_SRC_MASK,
800 osr->clk_src << NAU8325_CLK_DAC_SRC_SFT);
801
802 uint16_t val_len = 0;
803 switch (bits_per_sample) {
804 case 16:
805 val_len = NAU8325_I2S_DL_16;
806 break;
807 case 20:
808 val_len = NAU8325_I2S_DL_20;
809 break;
810 case 24:
811 val_len = NAU8325_I2S_DL_24;
812 break;
813 case 32:
814 val_len = NAU8325_I2S_DL_32;
815 break;
816 default:
817 AD_LOGI("[NAU8325] Invalid bit width: %u\n", bits_per_sample);
818 return false;
819 }
820
821 writeRegisterBits(NAU8325_R0D_I2S_PCM_CTRL1, NAU8325_I2S_DL_MASK, val_len);
822
823 AD_LOGI("[NAU8325] configureAudio(): fs=%u, mclk=%u, bits=%u\n", fs, mclk,
824 bits_per_sample);
825
826 return true;

877 void initRegisters() {
878 writeRegisterBits(NAU8325_R2C_ALC_CTRL1, NAU8325_ALC_MAXGAIN_MASK,
879 0x7 << NAU8325_ALC_MAXGAIN_SFT);
880 writeRegisterBits(
881 NAU8325_R2D_ALC_CTRL2,
882 NAU8325_ALC_DCY_MASK | NAU8325_ALC_ATK_MASK | NAU8325_ALC_HLD_MASK,
883 (0x5 << NAU8325_ALC_DCY_SFT) | (0x3 << NAU8325_ALC_ATK_SFT) |
884 (0x5 << NAU8325_ALC_HLD_SFT));
885
886 if (alc_enable) {
887 writeRegisterBits(NAU8325_R2E_ALC_CTRL3, NAU8325_ALC_EN, NAU8325_ALC_EN);
888 }
889
890 uint16_t clk_det_mask = NAU8325_CLKPWRUP_DIS | NAU8325_PWRUP_DFT;
891
892 if (clock_detection)
893 writeRegisterBits(NAU8325_R40_CLK_DET_CTRL, clk_det_mask, 0);
894 else
895 writeRegisterBits(NAU8325_R40_CLK_DET_CTRL, clk_det_mask,
896 NAU8325_CLKPWRUP_DIS);
897
898 if (clock_det_data)
899 writeRegisterBits(NAU8325_R40_CLK_DET_CTRL, NAU8325_APWRUP_EN,
900 NAU8325_APWRUP_EN);
901 else
902 writeRegisterBits(NAU8325_R40_CLK_DET_CTRL, NAU8325_APWRUP_EN, 0);
903
904 uint16_t vref_val = 0xFFFF;
905 if (dac_vref_microvolt == 1800000)
906 vref_val = 0 << NAU8325_DACVREFSEL_SFT;
907 else if (dac_vref_microvolt == 2700000)
908 vref_val = 1 << NAU8325_DACVREFSEL_SFT;
909 else if (dac_vref_microvolt == 2880000)
910 vref_val = 2 << NAU8325_DACVREFSEL_SFT;
911 else if (dac_vref_microvolt == 3060000)
912 vref_val = 3 << NAU8325_DACVREFSEL_SFT;
913
914 if (vref_val != 0xFFFF) {

986

988 uint8_t dataBuf[2] = {(uint8_t)((val >> 8) & 0xFF), (uint8_t)(val & 0xFF)};
989
990 return i2c_bus_write_bytes(this->i2c, i2c_addr, regBuf, sizeof(regBuf),

992 }

996 uint8_t regBuf[2] = {(uint8_t)((reg >> 8) & 0xFF), (uint8_t)(reg & 0xFF)};

1000 sizeof(dataBuf)) != RESULT_OK) {
1001 return false;
1002 }
1003
1004 value = ((uint16_t)dataBuf[0] << 8) | dataBuf[1];

1006 }
