74struct _coeff_div {
75 uint32_t mclk; /* mclk frequency */
76 uint32_t lrck; /* lrck */
77 uint8_t ss_ds;
78 uint8_t adc_div; /* adcclk divider */
79 uint8_t dll; /* dll_bypass */
80 uint8_t doubler; /* doubler enable */
81 uint8_t osr; /* adc osr */
82 uint8_t mclk_src; /* select mclk source */
83 uint32_t lrck_h; /* The high 4 bits of lrck */
84 uint32_t lrck_l; /* The low 8 bits of lrck */
85};

207 error_t init(codec_config_t* codec_cfg) {
208 // select microphones
209 error_t ret = setMicsForChannels(codec_cfg->i2s.channels);
210
211 ret |= writeReg(ES7210_RESET_REG00, 0xff);
212 ret |= writeReg(ES7210_RESET_REG00, 0x41);
213 ret |= writeReg(ES7210_CLOCK_OFF_REG01, 0x1f);
214 ret |= writeReg(ES7210_TIME_CONTROL0_REG09,
215 0x30); /* Set chip state cycle */
216 ret |= writeReg(ES7210_TIME_CONTROL1_REG0A,
217 0x30); /* Set power on state cycle */
218 ret |= writeReg(ES7210_ADC12_HPF2_REG23, 0x2a); /* Quick setup */
219 ret |= writeReg(ES7210_ADC12_HPF1_REG22, 0x0a);
220 ret |= writeReg(ES7210_ADC34_HPF2_REG20, 0x0a);
221 ret |= writeReg(ES7210_ADC34_HPF1_REG21, 0x2a);
222 /* Set master/slave audio interface */
223 I2SDefinition* i2s_cfg = &(codec_cfg->i2s);
224 switch (i2s_cfg->mode) {
225 case MODE_MASTER: /* MASTER MODE */
226 AD_LOGI("ES7210 in Master mode");
227 ret |= updateRegBit(ES7210_MODE_CONFIG_REG08, 0x01, 0x01);
228 /* Select clock source for internal mclk */
229 switch (getMclkSrc()) {
230 case FROM_PAD_PIN:
231 ret |= updateRegBit(ES7210_MASTER_CLK_REG03, 0x80, 0x00);
232 break;
233 case FROM_CLOCK_DOUBLE_PIN:
234 ret |= updateRegBit(ES7210_MASTER_CLK_REG03, 0x80, 0x80);
235 break;
236 default:
237 ret |= updateRegBit(ES7210_MASTER_CLK_REG03, 0x80, 0x00);
238 break;
239 }
240 break;
241 case MODE_SLAVE: /* SLAVE MODE */
242 AD_LOGI("ES7210 in Slave mode");
243 ret |= updateRegBit(ES7210_MODE_CONFIG_REG08, 0x01, 0x00);
244 break;
245 default:
246 ret |= updateRegBit(ES7210_MODE_CONFIG_REG08, 0x01, 0x00);
247 }
248 ret |= writeReg(ES7210_ANALOG_REG40,
249 0x43); /* Select power off analog, vdda = 3.3V, close
250 vx20ff, VMID select 5KΩ start */
251 ret |= writeReg(ES7210_MIC12_BIAS_REG41, 0x70); /* Select 2.87v */
252 ret |= writeReg(ES7210_MIC34_BIAS_REG42, 0x70); /* Select 2.87v */
253 ret |= writeReg(ES7210_OSR_REG07, 0x20);
254 ret |= writeReg(
255 ES7210_MAINCLK_REG02,
256 0xc1); /* Set the frequency division coefficient and use dll except
257 clock doubler, and need to set 0xc1 to clear the state */
258 ret |= configSample(i2s_cfg->rate);
259 ret |= micSelect(mic_select);
260 ret |= setGain(ES7210_GAIN_30DB);
261
262 // set REG12
263 uint8_t reg12 = 0;
264 if (i2s_cfg->signal_type == SIGNAL_TDM) {
265 reg12 = 0x02; // TDM I2S
266 }
267 ret |= writeReg(ES7210_SDP_INTERFACE2_REG12, reg12);
268
269 return RESULT_OK;
270 }

285 error_t ctrlStateActive(codec_mode_t mode, bool ctrl_state_active) {
286 static uint8_t regv;
287 error_t ret = RESULT_OK;
288 AD_LOGW("ES7210 only supports ADC mode");
289 regv = readReg(ES7210_CLOCK_OFF_REG01);
290 if ((regv != 0x7f) && (regv != 0xff)) {
291 regv = readReg(ES7210_CLOCK_OFF_REG01);
292 }
293 if (ctrl_state_active) {
294 AD_LOGI("The ES7210_CLOCK_OFF_REG01 value before stop is %x", regv);
295 ret |= start(regv);
296 } else {
297 AD_LOGW("The codec is about to stop");
298 regv = readReg(ES7210_CLOCK_OFF_REG01);
299 ret |= stop();
300 }
301 return ret;
302 }

305 error_t micSelect(es7210_input_mics_t mic) {
306 error_t ret = RESULT_OK;
307 mic_select = mic;
308 if (mic_select & (ES7210_INPUT_MIC1 | ES7210_INPUT_MIC2 |
309 ES7210_INPUT_MIC3 | ES7210_INPUT_MIC4)) {
310 for (int i = 0; i < 4; i++) {
311 ret |= updateRegBit(ES7210_MIC1_ES7210_GAIN_REG43 + i, 0x10, 0x00);
312 }
313 ret |= writeReg(ES7210_MIC12_POWER_REG4B, 0xff);
314 ret |= writeReg(ES7210_MIC34_POWER_REG4C, 0xff);
315 if (mic_select & ES7210_INPUT_MIC1) {
316 AD_LOGI("Enable ES7210_INPUT_MIC1");
317 ret |= updateRegBit(ES7210_CLOCK_OFF_REG01, 0x0b, 0x00);
318 ret |= writeReg(ES7210_MIC12_POWER_REG4B, 0x00);
319 ret |= updateRegBit(ES7210_MIC1_ES7210_GAIN_REG43, 0x10, 0x10);
320 }
321 if (mic_select & ES7210_INPUT_MIC2) {
322 AD_LOGI("Enable ES7210_INPUT_MIC2");
323 ret |= updateRegBit(ES7210_CLOCK_OFF_REG01, 0x0b, 0x00);
324 ret |= writeReg(ES7210_MIC12_POWER_REG4B, 0x00);
325 ret |= updateRegBit(ES7210_MIC2_ES7210_GAIN_REG44, 0x10, 0x10);
326 }
327 if (mic_select & ES7210_INPUT_MIC3) {
328 AD_LOGI("Enable ES7210_INPUT_MIC3");
329 ret |= updateRegBit(ES7210_CLOCK_OFF_REG01, 0x15, 0x00);
330 ret |= writeReg(ES7210_MIC34_POWER_REG4C, 0x00);
331 ret |= updateRegBit(ES7210_MIC3_ES7210_GAIN_REG45, 0x10, 0x10);
332 }
333 if (mic_select & ES7210_INPUT_MIC4) {
334 AD_LOGI("Enable ES7210_INPUT_MIC4");
335 ret |= updateRegBit(ES7210_CLOCK_OFF_REG01, 0x15, 0x00);
336 ret |= writeReg(ES7210_MIC34_POWER_REG4C, 0x00);
337 ret |= updateRegBit(ES7210_MIC4_ES7210_GAIN_REG46, 0x10, 0x10);
338 }
339 } else {
340 AD_LOGE("Microphone selection error");
341 return RESULT_FAIL;
342 }
343 return ret;
344 }

347 error_t setGain(es7210_gain_value_t gain) {
348 AD_LOGD("setGain:%d", gain);
349 error_t ret = RESULT_OK;
350 es7210_gain_value_t max_gain_vaule = static_cast<es7210_gain_value_t>(14);
351 if (gain < 0) {
352 gain = (es7210_gain_value_t)0;
353 } else if (gain > max_gain_vaule) {
354 gain = max_gain_vaule;
355 }
356 if (mic_select & ES7210_INPUT_MIC1) {
357 ret |= updateRegBit(ES7210_MIC1_ES7210_GAIN_REG43, 0x0f, gain);
358 }
359 if (mic_select & ES7210_INPUT_MIC2) {
360 ret |= updateRegBit(ES7210_MIC2_ES7210_GAIN_REG44, 0x0f, gain);
361 }
362 if (mic_select & ES7210_INPUT_MIC3) {
363 ret |= updateRegBit(ES7210_MIC3_ES7210_GAIN_REG45, 0x0f, gain);
364 }
365 if (mic_select & ES7210_INPUT_MIC4) {
366 ret |= updateRegBit(ES7210_MIC4_ES7210_GAIN_REG46, 0x0f, gain);
367 }
368 return ret;
369 }

372 error_t getGain(void) {
373 int regv = 0;
374 uint8_t gain_value;
375 if (mic_select & ES7210_INPUT_MIC1) {
376 regv = readReg(ES7210_MIC1_ES7210_GAIN_REG43);
377 } else if (mic_select & ES7210_INPUT_MIC2) {
378 regv = readReg(ES7210_MIC2_ES7210_GAIN_REG44);
379 } else if (mic_select & ES7210_INPUT_MIC3) {
380 regv = readReg(ES7210_MIC3_ES7210_GAIN_REG45);
381 } else if (mic_select & ES7210_INPUT_MIC4) {
382 regv = readReg(ES7210_MIC4_ES7210_GAIN_REG46);
383 } else {
384 AD_LOGE("No MIC selected");
385 return RESULT_FAIL;
386 }
387 if (regv == RESULT_FAIL) {
388 return regv;
389 }
390 gain_value = (regv & 0x0f); /* Retain the last four bits for gain */
391 AD_LOGI("GET: gain_value:%d", gain_value);
392 return gain_value;
393 }

428 error_t updateRegBit(uint8_t reg_addr, uint8_t update_bits, uint8_t data) {
429 uint8_t regv;
430 regv = readReg(reg_addr);
431 regv = (regv & (~update_bits)) | (update_bits & data);
432 return writeReg(reg_addr, regv);

435 int getCoeff(uint32_t mclk, uint32_t lrck) {
436 for (unsigned i = 0;
437 i < (sizeof(es7210_coeff_div) / sizeof(es7210_coeff_div[0])); i++) {
438 if (es7210_coeff_div[i].lrck == lrck && es7210_coeff_div[i].mclk == mclk)
439 return i;
440 }
441 return -1;
442 }

446 error_t configSample(samplerate_t sample) {
447 uint8_t regv;
448 int coeff;
449 int sample_fre = 0;
450 int mclk_fre = 0;
451 error_t ret = RESULT_OK;
452 switch (sample) {
453 case RATE_8K:
454 sample_fre = 8000;
455 break;
456 case RATE_11K:
457 sample_fre = 11025;
458 break;
459 case RATE_16K:
460 sample_fre = 16000;
461 break;
462 case RATE_22K:
463 sample_fre = 22050;
464 break;
465 case RATE_24K:
466 sample_fre = 24000;
467 break;
468 case RATE_32K:
469 sample_fre = 32000;
470 break;
471 case RATE_44K:
472 sample_fre = 44100;
473 break;
474 case RATE_48K:
475 sample_fre = 48000;
476 break;
477 default:
478 AD_LOGE("Unable to configure sample rate %dHz", sample_fre);
479 break;
480 }
481 mclk_fre = sample_fre * MCLK_DIV_FRE;
482 coeff = getCoeff(mclk_fre, sample_fre);
483 if (coeff < 0) {
484 AD_LOGE("Unable to configure sample rate %dHz with %dHz MCLK", sample_fre,
485 mclk_fre);
486 return RESULT_FAIL;
487 }
488 /* Set clock parammeters */
489 if (coeff >= 0) {
490 /* Set adc_div & doubler & dll */
491 regv = readReg(ES7210_MAINCLK_REG02) & 0x00;
492 regv |= es7210_coeff_div[coeff].adc_div;
493 regv |= es7210_coeff_div[coeff].doubler << 6;
494 regv |= es7210_coeff_div[coeff].dll << 7;
495 ret |= writeReg(ES7210_MAINCLK_REG02, regv);
496 /* Set osr */
497 regv = es7210_coeff_div[coeff].osr;
498 ret |= writeReg(ES7210_OSR_REG07, regv);
499 /* Set lrck */
500 regv = es7210_coeff_div[coeff].lrck_h;
501 ret |= writeReg(ES7210_LRCK_DIVH_REG04, regv);
502 regv = es7210_coeff_div[coeff].lrck_l;
503 ret |= writeReg(ES7210_LRCK_DIVL_REG05, regv);
504 }
505 return ret;
506 }

527 error_t configFmt(i2s_format_t fmt) {
528 error_t ret = RESULT_OK;
529 uint8_t adc_iface = 0;
530
531 adc_iface = readReg(ES7210_SDP_INTERFACE1_REG11);
532 adc_iface &= 0xfc;
533
534 switch (fmt) {
535 case I2S_NORMAL:
536 AD_LOGD("ES7210 in I2S Format");
537 adc_iface |= 0x00;
538 break;
539 case I2S_LEFT:
540 case I2S_RIGHT:
541 AD_LOGD("ES7210 in LJ Format");
542 adc_iface |= 0x01;
543 break;
544 case I2S_DSP:
545 if (I2S_DSP_MODE) {
546 AD_LOGD("ES7210 in DSP-A Format");
547 adc_iface |= 0x13;
548 } else {
549 AD_LOGD("ES7210 in DSP-B Format");
550 adc_iface |= 0x03;
551 }
552 break;
553 default:
554 adc_iface &= 0xfc;
555 break;
556 }
557 ret |= writeReg(ES7210_SDP_INTERFACE1_REG11, adc_iface);
558 return ret;
559 }
560
561 error_t setBits(sample_bits_t bits) {
562 error_t ret = RESULT_OK;
563 uint8_t adc_iface = 0;
564 adc_iface = readReg(ES7210_SDP_INTERFACE1_REG11);
565 adc_iface &= 0x1f;
566 switch (bits) {
567 case BIT_LENGTH_16BITS:
568 adc_iface |= 0x60;
569 break;
570 case BIT_LENGTH_24BITS:
571 adc_iface |= 0x00;
572 break;
573 case BIT_LENGTH_32BITS:
574 adc_iface |= 0x80;
575 break;
576 default:
577 adc_iface |= 0x60;
578 break;
579 }
580 ret |= writeReg(ES7210_SDP_INTERFACE1_REG11, adc_iface);
581 return ret;
582 }
583
584 error_t start(uint8_t clock_reg_value) {
585 error_t ret = RESULT_OK;
586 ret |= writeReg(ES7210_CLOCK_OFF_REG01, clock_reg_value);
587 ret |= writeReg(ES7210_POWER_DOWN_REG06, 0x00);
588 ret |= writeReg(ES7210_ANALOG_REG40, 0x43);
589 ret |= writeReg(ES7210_MIC1_POWER_REG47, 0x00);
590 ret |= writeReg(ES7210_MIC2_POWER_REG48, 0x00);
591 ret |= writeReg(ES7210_MIC3_POWER_REG49, 0x00);
592 ret |= writeReg(ES7210_MIC4_POWER_REG4A, 0x00);
593 ret |= micSelect(mic_select);
594 return ret;
595 }
596
597 error_t stop(void) {
598 error_t ret = RESULT_OK;
599 ret |= writeReg(ES7210_MIC1_POWER_REG47, 0xff);
600 ret |= writeReg(ES7210_MIC2_POWER_REG48, 0xff);
601 ret |= writeReg(ES7210_MIC3_POWER_REG49, 0xff);
602 ret |= writeReg(ES7210_MIC4_POWER_REG4A, 0xff);
603 ret |= writeReg(ES7210_MIC12_POWER_REG4B, 0xff);
604 ret |= writeReg(ES7210_MIC34_POWER_REG4C, 0xff);
605 ret |= writeReg(ES7210_ANALOG_REG40, 0xc0);
606 ret |= writeReg(ES7210_CLOCK_OFF_REG01, 0x7f);
607 ret |= writeReg(ES7210_POWER_DOWN_REG06, 0x07);
608 return ret;
609 }
610
611 protected:
612 i2c_bus_handle_t i2c_handle = nullptr;
613 int i2c_addr = ES7210_ADDR;
614 es7210_input_mics_t mic_select = static_cast<es7210_input_mics_t>(
615 ES7210_INPUT_MIC1 | ES7210_INPUT_MIC2); /* Number of microphones */
616};
