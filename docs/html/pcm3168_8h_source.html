142 inline uint8_t bit_set_to(uint8_t number, uint8_t n, bool x) {
143 return (number & ~((uint8_t)1 << n)) | ((uint8_t)x << n);
144 }

146 // convert level to volume byte, Table 21, page 45
147 uint32_t volumebyte(float level) {
148 if (level >= 1.0f) return 255;
149 if (level <= 1.0e-5f) return 54; // below -100dB, mute
150 // 0.5dB steps: 0dB attenuation is 255
151 return 255 + roundf(log10f(level) * 40.0f);
152 }
153
154 // convert level to input gain, Table 20, page 50
155 int32_t inputlevelbyte(float level) {
156 if (level > 10.0) return 255;
157 if (level < 1.0e-5f) return 15;
158 return 215 + roundf(log10f(level) * 40.0f);
159 }

161 bool volumeInteger(uint32_t n) { return write(DAC_ATTENUATION_ALL, n); }

163 bool volumeInteger(int channel, uint32_t n) {
164 bool rv = false;
165
166 if (0 != channel && channel <= DAC_CHANNELS_MAX)
167 rv = write(DAC_ATTENUATION_BASE + channel - 1, n);
168
169 return rv;
170 }
171
172 bool inputLevelInteger(int32_t n) {
173 uint8_t levels[ADC_CHANNELS_MAX];
174
175 for (int i = 0; i < ADC_CHANNELS_MAX; i++) levels[i] = n;
176
177 return write(ADC_ATTENUATION_BASE, levels, ADC_CHANNELS_MAX);
178 }
