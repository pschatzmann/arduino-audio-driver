253 bool begin(TAS2563SampleRate rate = TAS2563SampleRate::Rate48kHz,
254 uint8_t word_size = 16,
255 TAS2563Format format = TAS2563Format::I2S) {
256 bool rc = true;
257
258 rc &= softReset();
259
260 uint8_t rev_id = 0;
261 readReg(REG_REV_ID, rev_id);
262
263 // shut down before re-configuring
264 rc &= updateReg(REG_PWR_CTL, PWR_MODE_MASK,
265 (uint8_t)TAS2563PowerMode::Shutdown);
266
267 rc &= setFormat(rate, word_size, format);
268
269 // default amplifier level (16.0 dBV)
270 rc &= setAmpLevel(AMP_LEVEL_16_0DBV);
271
272 return rc;
273 }

285 bool setFormat(TAS2563SampleRate rate, uint8_t word_size = 16,
286 TAS2563Format format = TAS2563Format::I2S) {
287 bool rc = true;
288 uint8_t tdm_cfg0 = 0, tdm_cfg1 = 0, tdm_cfg2 = 0;
289
290 tdm_cfg0 |= (uint8_t)((uint8_t)rate << SAMP_RATE_SHIFT) & SAMP_RATE_MASK;
291
292 switch (word_size) {
293 case 16:
294 tdm_cfg2 |= RX_WLEN_16BITS << RX_WLEN_SHIFT;
295 tdm_cfg2 |= RX_SLEN_16BITS;
296 break;
297 case 24:
298 tdm_cfg2 |= RX_WLEN_24BITS << RX_WLEN_SHIFT;
299 tdm_cfg2 |= RX_SLEN_32BITS;
300 break;
301 case 32:
302 tdm_cfg2 |= RX_WLEN_32BITS << RX_WLEN_SHIFT;
303 tdm_cfg2 |= RX_SLEN_32BITS;
304 break;
305 default:
306 return false;
307 }
308
309 switch (format) {
310 case TAS2563Format::I2S:
311 tdm_cfg0 |= FRAME_START;
312 tdm_cfg1 |= (uint8_t)(1U << RX_OFFSET_SHIFT) & RX_OFFSET_MASK;
313 break;
314 case TAS2563Format::LeftJustified:
315 tdm_cfg0 &= (uint8_t)~FRAME_START;
316 tdm_cfg1 &= (uint8_t)~RX_OFFSET_MASK;
317 break;
318 }
319
320 tdm_cfg2 |= (RX_SCFG_MONO_LEFT << RX_SCFG_SHIFT) & RX_SCFG_MASK;
321
322 rc &= writeReg(REG_TDM_CFG0, tdm_cfg0);
323 rc &= writeReg(REG_TDM_CFG1, tdm_cfg1);
324 rc &= updateReg(REG_TDM_CFG2, CFG2_CONFIG_MASK, tdm_cfg2);
325
326 return rc;
327 }

336 bool setVolume(int volume) override {
337 uint8_t amp_level;
338 if (volume <= 0) {
339 amp_level = AMP_LEVEL_8_5DBV;
340 } else if (volume >= 100) {
341 amp_level = AMP_LEVEL_22_0DBV;
342 } else {
343 amp_level = (uint8_t)(AMP_LEVEL_8_5DBV +
344 (volume * (AMP_LEVEL_22_0DBV - AMP_LEVEL_8_5DBV)) /
345 100);
346 }
347 return setAmpLevel(amp_level);
348 }
