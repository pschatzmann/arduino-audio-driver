1042 bool setupPll(uint32_t mclk_hz, wm8960_adc_dac_sample_rate_t sample_rate) {
1043 AD_LOGD("setupPll");
1044 bool result;
1045 uint8_t PLLN;
1046 uint32_t PLLK;
1047 bool use_prescale = false;
1048 uint32_t sys_clk_hz;
1049
1050 switch (sample_rate) {
1051 case WM8960_ADC_DAC_SAMPLE_RATE_48_KHZ:
1052 case WM8960_ADC_DAC_SAMPLE_RATE_32_KHZ:
1053 case WM8960_ADC_DAC_SAMPLE_RATE_24_KHZ:
1054 case WM8960_ADC_DAC_SAMPLE_RATE_16_KHZ:
1055 case WM8960_ADC_DAC_SAMPLE_RATE_12_KHZ:
1056 case WM8960_ADC_DAC_SAMPLE_RATE_8_KHZ:
1057 default:
1058 sys_clk_hz = _WM8960_SYSCLK_FREQ_12288000_HZ;
1059 break;
1060 case WM8960_ADC_DAC_SAMPLE_RATE_44_1_KHZ:
1061 case WM8960_ADC_DAC_SAMPLE_RATE_22_05_KHZ:
1062 case WM8960_ADC_DAC_SAMPLE_RATE_11_025_KHZ:
1063 case WM8960_ADC_DAC_SAMPLE_RATE_8_018_KHZ:
1064 sys_clk_hz = _WM8960_SYSCLK_FREQ_11289600_HZ;
1065 break;
1066 }
1067
1068 uint32_t f2 = (4 * 2 * sys_clk_hz);
1069 uint32_t f1 = mclk_hz;
1070 float R = (float)f2 / (float)f1;
1071 if ((R <= 5) || ((R * 2 - 8) < (8 - R))) {
1072 use_prescale = true;
1073 R *= (float)2;
1074 }
1075 if ((R <= 5) || (R >= 13)) {
1076 return false;
1077 }
1078
1079 PLLN = (uint8_t)R;
1080 PLLK = (uint32_t)((float)0x1000000 * (R - (float)PLLN));
1081 uint16_t prescale_mask = (use_prescale) ? WM8960_PLL_N_PLLPRESCALE_EN
1082 : WM8960_PLL_N_PLLPRESCALE_DI;
1083
1084 result = write(WM8960_REG_PLL_N,
1085 PLLN | prescale_mask | WM8960_PLL_N_SDM_FRAC);
1086 if (!result) return result;
1087
1088 result = write(WM8960_REG_PLL_K1, PLLK >> 16 & 0xFF);
1089 if (!result) return result;
1090
1091 result = write(WM8960_REG_PLL_K2, PLLK >> 8 & 0xFF);
1092 if (!result) return result;
1093
1094 result = write(WM8960_REG_PLL_K3, PLLK & 0xFF);
1095 if (!result) return result;
1096
1097 result = set(WM8960_REG_PWR_MGMT2, WM8960_PWR_MGMT2_PLL_EN_UP);
1098 return result;
1099 }
