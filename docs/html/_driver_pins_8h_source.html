184 virtual bool begin() {

282 void setupPinMode() {
283 gpio.begin(*this);
284 AD_LOGD("DriverPins::setupPinMode");
285 // setup pins
286 for (auto& tmp : pins) {
287 if (tmp.pin != -1) {
288 if (!hasConflict(tmp.pin)) {
289 AD_LOGD("pinMode for %d", tmp.pin);
290 switch (tmp.pin_logic) {
291 case PinLogic::InputActiveHigh:
292 gpio.pinMode(tmp.pin, INPUT);
293 break;
294 case PinLogic::InputActiveLow:
295 gpio.pinMode(tmp.pin, INPUT_PULLUP);
296 break;
297 case PinLogic::Input:
298 gpio.pinMode(tmp.pin, INPUT);
299 break;
300 case PinLogic::Output:
301 gpio.pinMode(tmp.pin, OUTPUT);
302 break;
303 default:
304 // do nothing
305 break;
306 }
307 } else {
308 AD_LOGW("Pin '%d' not set up because of conflict", tmp.pin);
309 tmp.active = false;
310 }
311 } else {
312 AD_LOGD("Pin is -1");
313 }
314 }
315 }
316
317 bool hasConflict(int pin) {
318 return hasSPIConflict(pin) || hasI2CConflict(pin);
319 }
320
321 bool hasSPIConflict(int pin) {
322 if (sd_active) {
323 auto sd = getSPIPins(PinFunction::SD);
324 if (!sd) return false;
325 PinsSPI spi = sd.value();
326 return spi.cs == pin || spi.clk == pin || spi.miso == pin ||
327 spi.mosi == pin;
328 }
329 return false;
330 }
331
332 bool hasI2CConflict(int pin) {
333 for (auto& i2c_entry : i2c) {
334 if (i2c_entry.scl == pin || i2c_entry.sda == pin) {
335 return true;
336 }
337 }
338 return false;
339 }

349 bool result = false;
350#if defined(ESP32) && defined(ARDUINO)
351#if SOC_TOUCH_SENSOR_SUPPORTED
352 auto pin_opt = getPin(PinFunction::KEY, key);
353 if (!pin_opt) return false;
354 auto pin = pin_opt.value();
355 int value = touchRead(pin.pin);
356 result = value <= touch_limit;
357 if (result) {
358 // retry to confirm reading
359 value = touchRead(pin.pin);
360 result = value <= touch_limit;
361 }
362#endif
363#endif
