285 void setupPinMode() {
286 gpio.begin(*this);
287 AD_LOGD("DriverDeviceInfo::setupPinMode");
288 // setup pins
289 for (auto& tmp : pins) {
290 if (tmp.pin != -1) {
291 if (!hasConflict(tmp.pin)) {
292 AD_LOGD("pinMode for %d", tmp.pin);
293 switch (tmp.pin_logic) {
294 case PinLogic::InputActiveHigh:
295 gpio.pinMode(tmp.pin, INPUT);
296 break;
297 case PinLogic::InputActiveLow:
298 gpio.pinMode(tmp.pin, INPUT_PULLUP);
299 break;
300 case PinLogic::Input:
301 gpio.pinMode(tmp.pin, INPUT);
302 break;
303 case PinLogic::Output:
304 gpio.pinMode(tmp.pin, OUTPUT);
305 break;
306 default:
307 // do nothing
308 break;
309 }
310 } else {
311 AD_LOGW("Pin '%d' not set up because of conflict", tmp.pin);
312 tmp.active = false;
313 }
314 } else {
315 AD_LOGD("Pin is -1");
316 }
317 }
318 }

320 bool hasConflict(int pin) {
321 return hasSPIConflict(pin) || hasI2CConflict(pin);
322 }
323
324 bool hasSPIConflict(int pin) {
325 if (sd_active) {
326 auto sd = getSPIPins(PinFunction::SD);
327 if (!sd) return false;
328 InfoSPI spi = sd.value();
329 return spi.cs == pin || spi.clk == pin || spi.miso == pin ||
330 spi.mosi == pin;
331 }
332 return false;
333 }
334
335 bool hasI2CConflict(int pin) {
336 for (auto& i2c_entry : i2c) {
337 if (i2c_entry.scl == pin || i2c_entry.sda == pin) {
338 return true;
339 }
340 }
341 return false;
342 }

352class DriverTouchClass : public DriverDeviceInfo {
353 bool isKeyPressed(uint8_t key) override {
354 bool result = false;
355#if defined(ESP32) && defined(ARDUINO)
356#if SOC_TOUCH_SENSOR_SUPPORTED
357 auto pin_opt = getPin(PinFunction::KEY, key);
358 if (!pin_opt) return false;
359 auto pin = pin_opt.value();
360 int value = touchRead(pin.pin);
361 result = value <= touch_limit;
362 if (result) {
363 // retry to confirm reading
364 value = touchRead(pin.pin);
365 result = value <= touch_limit;
366 }
367#endif
368#endif
369 return result;
370 }
