138 uint32_t init(uint16_t outputDevice, uint8_t volume, uint32_t audioFreq) {
139 uint32_t counter = 0;
140
141 /* Keep Codec powered OFF */
142 counter += writeReg(CS43L22_REG_POWER_CTL1, 0x01);
143
144 /*Save Output device for mute ON/OFF procedure*/
145 switch (outputDevice) {
146 case OUTPUT_DEVICE_SPEAKER:
147 output_dev = 0xFA;
148 break;
149
150 case OUTPUT_DEVICE_HEADPHONE:
151 output_dev = 0xAF;
152 break;
153
154 case OUTPUT_DEVICE_BOTH:
155 output_dev = 0xAA;
156 break;
157
158 case OUTPUT_DEVICE_AUTO:
159 output_dev = 0x05;
160 break;
161
162 default:
163 output_dev = 0x05;
164 break;
165 }
166
167 counter += writeReg(CS43L22_REG_POWER_CTL2, output_dev);
168
169 /* Clock configuration: Auto detection */
170 counter += writeReg(CS43L22_REG_CLOCKING_CTL, 0x81);
171
172 /* Set the Slave Mode and the audio Standard */
173 counter += writeReg(CS43L22_REG_INTERFACE_CTL1, CODEC_STANDARD);
174
175 /* Set the Master volume */
176 counter += setVolume(volume);
177
178 /* If the Speaker is enabled, set the Mono mode and volume attenuation level
179 */
180 if (outputDevice != OUTPUT_DEVICE_HEADPHONE) {
181 /* Set the Speaker Mono mode */
182 counter += writeReg(CS43L22_REG_PLAYBACK_CTL2, 0x06);
183
184 /* Set the Speaker attenuation level */
185 counter += writeReg(CS43L22_REG_SPEAKER_A_VOL, 0x00);
186 counter += writeReg(CS43L22_REG_SPEAKER_B_VOL, 0x00);
187 }
188
189 /* Additional configuration for the CODEC. These configurations are done to
190 reduce the time needed for the Codec to power off. If these configurations are
191 removed, then a long delay should be added between powering off the Codec and
192 switching off the I2S peripheral MCLK clock (which is the operating clock for
193 Codec). If this delay is not inserted, then the codec will not shut down
194 properly and it results in high noise after shut down. */
195
196 /* Disable the analog soft ramp */
197 counter += writeReg(CS43L22_REG_ANALOG_ZC_SR_SETT, 0x00);
198 /* Disable the digital soft ramp */
199 counter += writeReg(CS43L22_REG_MISC_CTL, 0x04);
200 /* Disable the limiter attack level */
201 counter += writeReg(CS43L22_REG_LIMIT_CTL1, 0x00);
202 /* Adjust Bass and Treble levels */
203 counter += writeReg(CS43L22_REG_TONE_CTL, 0x0F);
204 /* Adjust PCM volume level */
205 counter += writeReg(CS43L22_REG_PCMA_VOL, 0x0A);
206 counter += writeReg(CS43L22_REG_PCMB_VOL, 0x0A);
207
208 /* Return communication control value */
209 return counter;
210 }

228 counter += writeReg(CS43L22_REG_MISC_CTL, 0x06);

233 /* Power on the Codec */

275 uint32_t stop(uint32_t cmd) {
276 uint32_t counter = 0;
277
278 /* Mute the output first */
279 counter += setMute(AUDIO_MUTE_ON);
280
281 /* Disable the digital soft ramp */

285 counter += writeReg(CS43L22_REG_POWER_CTL1, 0x9F);

294 uint32_t setVolume(uint8_t volume) {
295 uint32_t counter = 0;
296 uint8_t convertedvol = cs43l22VolumeConvert(volume);
297
298 if (convertedvol > 0xE6) {
299 /* Set the Master volume */
300 counter += writeReg(CS43L22_REG_MASTER_A_VOL, convertedvol - 0xE7);
301 counter += writeReg(CS43L22_REG_MASTER_B_VOL, convertedvol - 0xE7);
302 } else {
303 /* Set the Master volume */
304 counter += writeReg(CS43L22_REG_MASTER_A_VOL, convertedvol + 0x19);
305 counter += writeReg(CS43L22_REG_MASTER_B_VOL, convertedvol + 0x19);
306 }
307
308 return counter;
309 }

317 uint32_t setMute(uint32_t cmd) {
318 uint32_t counter = 0;
319
320 /* Set the Mute mode */
321 if (cmd == AUDIO_MUTE_ON) {
322 counter += writeReg(CS43L22_REG_POWER_CTL2, 0xFF);
323 counter += writeReg(CS43L22_REG_HEADPHONE_A_VOL, 0x01);
324 counter += writeReg(CS43L22_REG_HEADPHONE_B_VOL, 0x01);
325 } else /* AUDIO_MUTE_OFF Disable the Mute */
326 {
327 counter += writeReg(CS43L22_REG_HEADPHONE_A_VOL, 0x00);
328 counter += writeReg(CS43L22_REG_HEADPHONE_B_VOL, 0x00);
329 counter += writeReg(CS43L22_REG_POWER_CTL2, output_dev);
330 }
331 return counter;
332 }

338 uint32_t setOutputMode(uint8_t output) {
339 uint32_t counter = 0;
340
341 switch (output) {
342 case OUTPUT_DEVICE_SPEAKER:
343 counter += writeReg(CS43L22_REG_POWER_CTL2,
344 0xFA); /* SPK always ON & HP always OFF */
345 output_dev = 0xFA;
346 break;
347
348 case OUTPUT_DEVICE_HEADPHONE:
349 counter += writeReg(CS43L22_REG_POWER_CTL2,
350 0xAF); /* SPK always OFF & HP always ON */
351 output_dev = 0xAF;
352 break;
353
354 case OUTPUT_DEVICE_BOTH:
355 counter += writeReg(CS43L22_REG_POWER_CTL2,
356 0xAA); /* SPK always ON & HP always ON */
357 output_dev = 0xAA;
358 break;
359
360 case OUTPUT_DEVICE_AUTO:
361 counter += writeReg(CS43L22_REG_POWER_CTL2,
362 0x05); /* Detect the HP or the SPK automatically */
363 output_dev = 0x05;
364 break;
365
366 default:
367 counter += writeReg(CS43L22_REG_POWER_CTL2,
368 0x05); /* Detect the HP or the SPK automatically */
369 output_dev = 0x05;
370 break;
371 }
372 return counter;
373 }
