83 error_t init(codec_config_t* cfg) {
84 if (initialized) {
85 AD_LOGW("The SGTL5000 has already been initialized");
86 return RESULT_OK;
87 }
88 initialized = true;
89
90 // VDDD is externally driven (typical for the common breakout boards)
91 writeReg(Reg::AnaPower, 0x4060);
92 // VDDA & VDDIO both over 3.1V
93 writeReg(Reg::LinregCtrl, 0x006C);
94 // VAG=1.575V, normal ramp, +12.5% bias current
95 writeReg(Reg::RefCtrl, 0x01F2);
96 // LO_VAGCNTRL=1.65V, OUT_CURRENT=0.54mA
97 writeReg(Reg::LineOutCtrl, 0x0F22);
98 // allow up to 125mA on the short detectors
99 writeReg(Reg::ShortCtrl, 0x4446);
100 // enable zero cross detectors
101 ana_ctrl = 0x0137;
102 writeReg(Reg::AnaCtrl, ana_ctrl);
103
104 // power up lineout, headphone, adc, dac
105 writeReg(Reg::AnaPower, 0x40FF);
106 // power up all digital blocks (I2S in/out, DAC, ADC, DAP)
107 writeReg(Reg::DigPower, 0x0073);
108 delayMs(400);
109
110 // default line out level ~1.3Vp-p
111 writeReg(Reg::LineOutVol, 0x1D1D);
112
113 // route I2S_IN -> DAC and ADC -> I2S_OUT
114 writeReg(Reg::SssCtrl, 0x0010);
115 // unmute the DAC
116 writeReg(Reg::AdcDacCtrl, 0x0000);
117 // digital gain 0dB
118 writeReg(Reg::DacVol, 0x3C3C);
119
120 // configure the digital audio interface and sample rate
121 configI2S(CODEC_MODE_BOTH, &cfg->i2s);
122
123 // set the default headphone/lineout volume and unmute
124 setVoiceVolume(70);
125
126 return RESULT_OK;
127 }

145 error_t configI2S(codec_mode_t mode, I2SDefinition* iface) {
146 uint16_t i2s_ctrl = 0;
147
148 switch (iface->fmt) {
149 case I2S_LEFT:
150 i2s_ctrl |= (I2SModeI2SOrLeftJustified << 2) | (1 << 1); // LRALIGN=1
151 break;
152 case I2S_RIGHT:
153 i2s_ctrl |= (I2SModeRightJustified << 2);
154 break;
155 case I2S_DSP:
156 i2s_ctrl |= (I2SModePcm << 2);
157 break;
158 case I2S_NORMAL:
159 default:
160 i2s_ctrl |= (I2SModeI2SOrLeftJustified << 2); // LRALIGN=0
161 break;
162 }
163
164 switch (iface->bits) {
165 case BIT_LENGTH_20BITS:
166 i2s_ctrl |= (0x2 << 4);
167 break;
168 case BIT_LENGTH_24BITS:
169 i2s_ctrl |= (0x1 << 4);
170 break;
171 case BIT_LENGTH_32BITS:
172 i2s_ctrl |= (0x0 << 4);
173 break;
174 case BIT_LENGTH_16BITS:
175 default:
176 i2s_ctrl |= (0x3 << 4);
177 break;
178 }
179
180 if (iface->mode == MODE_MASTER) {
181 i2s_ctrl |= (1 << 7);
182 }
183
184 writeReg(Reg::I2SCtrl, i2s_ctrl);
185 writeReg(Reg::ClkCtrl, getClkCtrl(iface->rate));
186 return RESULT_OK;
187 }

190 error_t setVoiceMute(bool enable) {
191 if (enable) {
192 ana_ctrl |= (MuteHeadphone | MuteLineOut);
193 } else {
194 ana_ctrl &= ~(MuteHeadphone | MuteLineOut);
195 }
196 return writeReg(Reg::AnaCtrl, ana_ctrl);
197 }

203 error_t setVoiceVolume(int volume) {
204 if (volume < 0) volume = 0;
205 if (volume > 100) volume = 100;
206 voice_volume = volume;
207
208 if (volume == 0) {
209 return setVoiceMute(true);
210 }
211
212 // 0x00 = +12dB ... 0x7F = -51.5dB, 0.5dB steps
213 uint16_t reg_value = 0x7F - ((volume * 0x7F) / 100);
214 error_t ret = writeReg(Reg::AnaHpCtrl, reg_value | (reg_value << 8));
215 ret |= setVoiceMute(false);
216 return ret;
217 }

226 uint16_t readReg(Reg reg) {
227 uint16_t reg_addr = static_cast<uint16_t>(reg);
228 uint8_t addr_bytes[2] = {(uint8_t)(reg_addr >> 8),
229 (uint8_t)(reg_addr & 0xFF)};
230 uint8_t data[2] = {0, 0};
231 i2c_bus_read_bytes(i2c_handle, i2c_addr, addr_bytes, 2, data, 2);
232 return ((uint16_t)data[0] << 8) | data[1];
233 }
234
236 error_t writeReg(Reg reg, uint16_t value) {
237 uint16_t reg_addr = static_cast<uint16_t>(reg);
238 uint8_t addr_bytes[2] = {(uint8_t)(reg_addr >> 8),
239 (uint8_t)(reg_addr & 0xFF)};
240 uint8_t data[2] = {(uint8_t)(value >> 8), (uint8_t)(value & 0xFF)};
241 return i2c_bus_write_bytes(i2c_handle, i2c_addr, addr_bytes, 2, data, 2);
242 }

249 uint16_t getClkCtrl(samplerate_t rate) {
250 uint16_t sys_fs;
251 switch (rate) {
252 case RATE_32K:
253 sys_fs = 0x0;
254 break;
255 case RATE_44K:
256 sys_fs = 0x1;
257 break;
258 case RATE_96K:
259 sys_fs = 0x3;
260 break;
261 case RATE_48K:
262 default:
263 sys_fs = 0x2;
264 break;
265 }
266 // RATE_MODE=0 (SYS_FS specifies the rate), MCLK_FREQ=0 (256*Fs)
267 return (sys_fs << 2);
268 }
