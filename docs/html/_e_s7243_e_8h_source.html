107 error_t ctrlStateActive(codec_mode_t mode, bool ctrl_state_active) {
108 error_t ret = RESULT_OK;
109 if (ctrl_state_active) {
110 ret |= writeReg(0xF9, 0x00);
111 ret |= writeReg(0x04, 0x01);
112 ret |= writeReg(0x17, 0x01);
113 ret |= writeReg(0x20, 0x10);
114 ret |= writeReg(0x21, 0x10);

116 ret |= writeReg(0x01, 0x3A);
117 ret |= writeReg(0x16, 0x3F);
118 ret |= writeReg(0x16, 0x00);
119 } else {
120 AD_LOGW("The codec going to stop");
121 ret |= writeReg(0x04, 0x02);
122 ret |= writeReg(0x04, 0x01);
123 ret |= writeReg(0xF7, 0x30);

133 ret |= writeReg(0x01, 0x00);

135 return ret;
