180 error_t init(codec_config_t* codec_cfg) {
181 error_t res = RESULT_OK;
182
183 res = writeReg(CHIP_AUDIO_RS, 0x123);
184 delayMs(1000);
185 if (res != RESULT_OK) {
186 AD_LOGE("reset failed!");
187 return res;
188 } else {
189 AD_LOGI("reset");
190 }
191 res |= writeReg(SPKOUT_CTRL, 0xe880);
192
193 // Enable the PLL from 256*44.1KHz MCLK source
194 res |= writeReg(PLL_CTRL1, 0x014f);
195 // res |= writeReg(PLL_CTRL2, 0x83c0);
196 res |= writeReg(PLL_CTRL2, 0x8600);
197
198 // Clocking system
199 res |= writeReg(SYSCLK_CTRL, 0x8b08);
200 res |= writeReg(MOD_CLK_ENA, 0x800c);
201 res |= writeReg(MOD_RST_CTRL, 0x800c);
202 res |= writeReg(I2S_SR_CTRL, 0x7000); // sample rate
203 // AIF config
204 res |= writeReg(I2S1LCK_CTRL, 0x8850); // BCLK/LRCK
205 res |= writeReg(I2S1_SDOUT_CTRL, 0xc000); //
206 res |= writeReg(I2S1_SDIN_CTRL, 0xc000);
207 res |= writeReg(I2S1_MXR_SRC, 0x2200); //
208
209 res |= writeReg(ADC_SRCBST_CTRL, 0xccc4);
210
211 res |= writeReg(ADC_SRC, getSrcValue(codec_cfg->input_device));
212 res |= writeReg(ADC_DIG_CTRL, 0x8000);
213 res |= writeReg(ADC_APC_CTRL, 0xbbc3);
214
215 // Path Configuration
216 res |= writeReg(DAC_MXR_SRC, 0xcc00);
217 res |= writeReg(DAC_DIG_CTRL, 0x8000);
218 res |= writeReg(OMIXER_SR, 0x0081);
219 res |= writeReg(OMIXER_DACA_CTRL, 0xf080); //}
220
221 //* Enable Speaker output
222 res |= writeReg(0x58, 0xeabd);
223
224 AD_LOGI("init done");
225 return res;
226 }

232 error_t ctrlStateActive(codec_mode_t mode, bool ctrlStateActive) {
233 int res = 0;
234 int es_mode_t = 0;
235
236 switch (mode) {
237 case CODEC_MODE_ENCODE:
238 es_mode_t = AC_MODULE_ADC;
239 break;
240 case CODEC_MODE_LINE_IN:
241 es_mode_t = AC_MODULE_LINE;
242 break;
243 case CODEC_MODE_DECODE:
244 es_mode_t = AC_MODULE_DAC;
245 break;
246 case CODEC_MODE_BOTH:
247 es_mode_t = AC_MODULE_ADC_DAC;
248 break;
249 default:
250 es_mode_t = AC_MODULE_DAC;
251 AD_LOGW("Codec mode not support, default is decode mode");
252 break;
253 }
254 if (!ctrlStateActive) {
255 res = stop((ac_module_t)es_mode_t);
256 } else {
257 res = start((ac_module_t)es_mode_t);
258 }
259 return res;
260 }

262 error_t configI2S(codec_mode_t mode, I2SDefinition* iface) {
263 error_t res = 0;
264 int bits = 0;
265 int fmat = 0;
266 int sample_fre = 0;
267 uint16_t regval;
268 switch (iface->bits) // 0x10
269 {
270 // case BIT_LENGTH_8BITS:
271 // bits = BIT_LENGTH_8_BITS;
272 // break;
273 case BIT_LENGTH_16BITS:
274 bits = BIT_LENGTH_16_BITS;
275 break;
276 case BIT_LENGTH_24BITS:
277 bits = BIT_LENGTH_24_BITS;
278 break;
279 default:
280 bits = BIT_LENGTH_16_BITS;
281 }
282
283 switch (iface->fmt) // 0x10
284 {
285 case I2S_NORMAL:
286 fmat = 0x0;
287 break;
288 case I2S_LEFT:
289 fmat = 0x01;
290 break;
291 case I2S_RIGHT:
292 fmat = 0x02;
293 break;
294 case I2S_DSP:
295 fmat = 0x03;
296 break;
297 default:
298 fmat = 0x00;
299 break;
300 }
301
302 switch (iface->rate) {
303 case RATE_8K:
304 sample_fre = 8000;
305 break;
306 case RATE_11K:
307 sample_fre = 11025;
308 break;
309 case RATE_16K:
310 sample_fre = 16000;
311 break;
312 case RATE_22K:
313 sample_fre = 22050;
314 break;
315 case RATE_24K:
316 sample_fre = 24000;
317 break;
318 case RATE_32K:
319 sample_fre = 32000;
320 break;
321 case RATE_44K:
322 sample_fre = 44100;
323 break;
324 case RATE_48K:
325 sample_fre = 48000;
326 break;
327 default:
328 sample_fre = 44100;
329 }
330 regval = readReg(I2S1LCK_CTRL);
331 regval &= 0xffc3;
332 regval |= (iface->mode << 15);
333 regval |= (bits << 4);
334 regval |= (fmat << 2);
335 res |= writeReg(I2S1LCK_CTRL, regval);
336 res |= writeReg(I2S_SR_CTRL, sample_fre);
337 return res;
338 }

358 error_t setSpkVolume(uint8_t volume) {
359 if (volume > 0x3f) volume = 0x3f;
360 volume = volume / 2;
361
362 uint16_t res;
363 error_t ret;
364
365 res = readReg(SPKOUT_CTRL);
366 res &= (~0x1f);
367 volume &= 0x1f;
368 res |= volume;
369 ret = writeReg(SPKOUT_CTRL, res);
370 return ret;
371 }

373 int getEarphVolume() {
374 int res;
375 res = readReg(HPOUT_CTRL);
376 return (res >> 4) & 0x3f;
377 }

379 error_t setEarphVolume(uint8_t volume) {
380 if (volume > 0x3f) volume = 0x3f;
381
382 uint16_t res, tmp;
383 error_t ret;
384 res = readReg(HPOUT_CTRL);
385 tmp = ~(0x3f << 4);
386 res &= tmp;
387 volume &= 0x3f;
388 res |= (volume << 4);
389 ret = writeReg(HPOUT_CTRL, res);
390 return ret;
391 }

393 error_t setOutputMixerGain(ac_output_mixer_gain_t gain,
394 ac_output_mixer_source_t source) {
395 uint16_t regval, temp, clrbit;
396 error_t ret;
397 regval = readReg(OMIXER_BST1_CTRL);
398 switch (source) {
399 case SRC_MIC1:
400 temp = (gain & 0x7) << 6;
401 clrbit = ~(0x7 << 6);
402 break;
403 case SRC_MIC2:
404 temp = (gain & 0x7) << 3;
405 clrbit = ~(0x7 << 3);
406 break;
407 case SRC_LINEIN:
408 temp = (gain & 0x7);
409 clrbit = ~0x7;
410 break;
411 default:
412 return -1;
413 }
414 regval &= clrbit;
415 regval |= temp;
416 ret = writeReg(OMIXER_BST1_CTRL, regval);
417 return ret;
418 }

441 error_t writeReg(uint8_t reg_add, uint16_t data) {
442 uint8_t send_buff[2];
443 send_buff[0] = (data >> 8) & 0xff;
444 send_buff[1] = data & 0xff;
445 return i2c_bus_write_bytes(i2c_handle, i2c_addr, &reg_add,
446 sizeof(reg_add), (uint8_t*)send_buff,

457 uint16_t val = 0;
458 uint8_t data_rd[2];
459 readI2C(i2c_addr, reg_addr, data_rd, 2);

522 error_t start(ac_module_t mode) {
523 error_t res = 0;
524 if (mode == AC_MODULE_LINE) {
525 res |= writeReg(0x51, 0x0408);
526 res |= writeReg(0x40, 0x8000);
527 res |= writeReg(0x50, 0x3bc0);
528 }
529 if (mode == AC_MODULE_ADC || mode == AC_MODULE_ADC_DAC ||
530 mode == AC_MODULE_LINE) {
531 // I2S1_SDOUT_CTRL
532 // res |= writeReg(PLL_CTRL2, 0x8120);
533 res |= writeReg(0x04, 0x800c);
534 res |= writeReg(0x05, 0x800c);
535 // res |= writeReg(0x06, 0x3000);
536 }
537 if (mode == AC_MODULE_DAC || mode == AC_MODULE_ADC_DAC ||
538 mode == AC_MODULE_LINE) {
539 //* Enable Headphoe output
540 res |= writeReg(OMIXER_DACA_CTRL, 0xff80);
541 res |= writeReg(HPOUT_CTRL, 0xc3c1);
542 res |= writeReg(HPOUT_CTRL, 0xcb00);
543 delayMs(100);
544 res |= writeReg(HPOUT_CTRL, 0xfbc0);
545
546 //* Enable Speaker output
547 res |= writeReg(SPKOUT_CTRL, 0xeabd);
548 delayMs(10);
549 setVoiceVolume(30);
550 }
551
552 return res;
553 }
554
555 error_t stop(ac_module_t mode) {
556 error_t res = 0;
557 res |= writeReg(HPOUT_CTRL, 0x01); // disable earphone
558 res |= writeReg(SPKOUT_CTRL, 0xe880); // disable speaker
559 return res;
560 }
