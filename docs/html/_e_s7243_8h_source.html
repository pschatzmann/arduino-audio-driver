46class ES7243 {
47 public:
48 static constexpr int MCLK_PULSES_NUMBER = 20;
49 ES7243() = default;
50
52 void setWire(i2c_bus_handle_t handle) { i2c_handle = handle; }
53
55 void setAddress(int addr) { i2c_addr = addr; }
56
58 void setMclkGpio(GpioPin gpio) { mclk_gpio = gpio; }

61 error_t init(codec_config_t* codec_cfg) {
62 error_t ret = RESULT_OK;
63 mclkActive(mclk_gpio);
64 ret |= writeReg(0x00, 0x01); // slave mode, software mode
65 ret |= writeReg(0x06, 0x00);
66 ret |= writeReg(0x05, 0x1B); // Mute ADC
67 ret |= writeReg(0x01, audio_format); // i2s -16bit
68 ret |= writeReg(0x08, 0x43); // enable AIN, PGA GAIN = 27DB
69 ret |= writeReg(0x05, 0x13); // un Mute ADC
70 if (ret) {
71 AD_LOGE("Es7243 initialize failed!");
72 return RESULT_FAIL;
73 }
74 return ret;
75 }
76
78 error_t deinit(void) { return setVoiceMute(true); }
79
81 error_t ctrlStateActive(codec_mode_t mode, bool ctrl_state_active) {
82 return setVoiceMute(!ctrl_state_active);
83 }
84
86 error_t configI2S(codec_mode_t mode, I2SDefinition* iface) {
87 // master mode not supported
88 if (iface->mode == MODE_MASTER) {
89 AD_LOGE("configI2S: Mode must be slave");
90 return RESULT_FAIL;
91 }
92
93 // set bits
94 switch (iface->bits) {
95 case BIT_LENGTH_16BITS:
96 audio_format |= 0b011 << 2;
97 break;
98 case BIT_LENGTH_24BITS:
99 audio_format |= 0b000 << 2;
100 break;
101 case BIT_LENGTH_32BITS:
102 audio_format |= 0b100 << 2;
103 break;
104 default:
105 AD_LOGE("configI2S: bits not supported %d", iface->bits);
106 return RESULT_FAIL;
107 }
108
109 switch (iface->fmt) {
110 case I2S_RIGHT:
111 case I2S_NORMAL:
112 audio_format |= 0b00;
113 break;
114 case I2S_LEFT:
115 audio_format |= 0b01;
116 break;
117 case I2S_DSP:
118 audio_format |= 0b11;
119 break;
120 default:
121 AD_LOGE("configI2S: fmt not supported %d", iface->fmt);
122 return RESULT_FAIL;
123 }
124
125 return writeReg(0x01, audio_format);
126 }
127
129 error_t setVoiceMute(bool mute) {
130 AD_LOGI("setVoiceMute: mute = %d", mute);
131 error_t ret = RESULT_OK;
132 if (mute) {
133 ret = writeReg(0x05, 0x1B);
134 } else {
135 ret = writeReg(0x05, 0x13);
136 }
137 return ret;
138 }
139
141 error_t setVoiceVolume(int volume) {
142 AD_LOGI("setVoiceVolume: %d", volume);
143 error_t ret = RESULT_OK;
144 if (volume > 100) {
145 volume = 100;
146 }
147 if (volume < 0) {
148 volume = 0;
149 }
150 actual_volume = volume;
151 switch (volume) {
152 case 0 ... 12:
153 ret |= writeReg(0x08, 0x11); // 1db
154 break;
155 case 13 ... 25:
156 ret |= writeReg(0x08, 0x13); // 3.5db
157 break;
158 case 26 ... 38:
159 ret |= writeReg(0x08, 0x21); // 18db
160 break;
161 case 39 ... 51:
162 ret |= writeReg(0x08, 0x23); // 20.5db
163 break;
164 case 52 ... 65:
165 ret |= writeReg(0x08, 0x05); // 22.5db
166 break;
167 case 66 ... 80:
168 ret |= writeReg(0x08, 0x41); // 24.5db
169 break;
170 case 81 ... 90:
171 ret |= writeReg(0x08, 0x07); // 25db
172 break;
173 case 91 ... 100:
174 ret |= writeReg(0x08, 0x43); // 27db
175 break;
176 default:
177 break;
178 }
179 return ret;
180 }
181
183 error_t getVoiceVolume(int* volume) {
184 *volume = actual_volume;
185 return RESULT_OK;
186 }
187 error_t writeReg(uint8_t reg_add, uint8_t data) {
188 return i2c_bus_write_bytes(i2c_handle, i2c_addr, &reg_add, sizeof(reg_add),
189 &data, sizeof(data));
190 }
191
192 /*
193 Before initializing es7243, it is necessary to output
194 mclk to es7243 to activate the I2C configuration.
195 So give some clocks to active es7243.
196 */
197 error_t mclkActive(GpioPin mclk_gpio) {
198 GPIO gpio;
199 gpio.pinMode(mclk_gpio, OUTPUT);
200 for (int i = 0; i < MCLK_PULSES_NUMBER; ++i) {
201 gpio.digitalWrite(mclk_gpio, 0);
202 delayMs(1);
203 gpio.digitalWrite(mclk_gpio, 1);
204 delayMs(1);
205 }
206 return RESULT_OK;
207 }
208
209 protected:
210 i2c_bus_handle_t i2c_handle = nullptr;
211 int i2c_addr = 0x13; // 0x26>>1;
212 GpioPin mclk_gpio = GPIO_UNDEFINED;
213 int actual_volume = 0;
214 uint8_t audio_format = 0x0C;
215};
