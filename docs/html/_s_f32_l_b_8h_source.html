311 bool setVolumeDb(int volume_db) {
312 if (volume_db < MIN_VOLUME_DB) volume_db = MIN_VOLUME_DB;
313 if (volume_db > MAX_VOLUME_DB) volume_db = MAX_VOLUME_DB;
314
315 uint32_t rough_vol = (volume_db - MIN_VOLUME_DB) / 6;
316 uint32_t fine_vol = ((volume_db - MIN_VOLUME_DB) % 6) << 1;
317
318 uint32_t value = (rough_vol << 8) | (fine_vol << 12);
319 uint32_t mask = DAC_CH_CFG_ROUGH_VOL_Msk | DAC_CH_CFG_FINE_VOL_Msk;
320 updateReg32(REG_DAC_CH0_CFG, mask, value);
321 last_fine_vol = fine_vol;
322 return true;
323 }

328 volume_percent = volume < 0 ? 0 : (volume > 100 ? 100 : volume);
329 int volume_db = MIN_VOLUME_DB +

331 return setVolumeDb(volume_db);

489 void configTxChannel(const DacClockConfig& clk) {
490 updateReg32(REG_CFG, CFG_ADC_EN_DLY_SEL_Msk, 3UL << 3);
491
492 uint32_t dac_cfg = (((uint32_t)clk.osr_sel) << 0) // OSR_SEL
493 | (0UL << 4) // OP_MODE
494 | (0UL << 7) // CLK_SRC_SEL (xtal)
495 | (((uint32_t)clk.clk_div) << 8); // CLK_DIV
496 write32(REG_DAC_CFG, dac_cfg);
497
498 constexpr uint32_t sinc_gain = 0x14D;
499 uint32_t ch0_cfg = DAC_CH_CFG_ENABLE | (2UL << 2) // DEM_MODE
500 | (6UL << 8) // ROUGH_VOL
501 | (0UL << 12) // FINE_VOL
502 | DAC_CH_CFG_DATA_FORMAT |
503 (sinc_gain << 17);
504 write32(REG_DAC_CH0_CFG, ch0_cfg);
505
506 uint32_t ch0_ext = DAC_CH_CFG_EXT_RAMP_EN | DAC_CH_CFG_EXT_RAMP_MODE |
507 DAC_CH_CFG_EXT_ZERO_ADJUST_EN |
508 (2UL << 3); // RAMP_INTERVAL
509 write32(REG_DAC_CH0_CFG_EXT, ch0_ext);
510
511 write32(REG_DAC_CH0_DEBUG, 0xFFUL); // BYPASS=0, DATA_OUT=0xFF
512 }
