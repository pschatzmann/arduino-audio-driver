229 bool begin(uint32_t sample_rate, uint8_t bits, codec_mode_t mode,
230 i2s_format_t fmt, bool is_master, uint8_t channels) override {
231 (void)channels;
232 DA7212Protocol protocol = DA7212Protocol::I2S;
233 switch (fmt) {
234 case I2S_LEFT:
235 protocol = DA7212Protocol::LeftJustified;
236 break;
237 case I2S_RIGHT:
238 protocol = DA7212Protocol::RightJustified;
239 break;
240 case I2S_DSP:
241 protocol = DA7212Protocol::DSPMode;
242 break;
243 default:
244 protocol = DA7212Protocol::I2S;
245 break;
246 }
247 return begin(protocol, sample_rate, bits, is_master,
248 mode & CODEC_MODE_DECODE, mode & CODEC_MODE_ENCODE);

251 bool begin(DA7212Protocol protocol = DA7212Protocol::I2S,
252 uint32_t sample_rate = 44100, uint8_t word_size = 16,
253 bool master = false, bool playback = true,
254 bool capture = true) {
255 bool rc = true;
256
257 rc &= softReset();
258
259 /* DAI right/left output stream comes from ADC right/left */
260 rc &= writeReg(REG_DIG_ROUTING_DAI,
261 DIG_ROUTING_DAI_R_SRC_ADC_RIGHT |
262 DIG_ROUTING_DAI_L_SRC_ADC_LEFT);
263
264 /* Default sample rate 16kHz, overwritten below */
265 rc &= writeReg(REG_SR, (uint8_t)DA7212SampleRate::Rate16kHz);
266
267 /* Enable voltage reference and bias */
268 rc &= writeReg(REG_REFERENCES, REFERENCES_BIAS_EN_MASK);
269
270 /* Keep PLL disabled, use MCLK as system clock */
271 rc &= writeReg(REG_PLL_FRAC_TOP, 0);
272 rc &= writeReg(REG_PLL_FRAC_BOT, 0);
273 rc &= writeReg(REG_PLL_INTEGER, PLL_FBDIV_INTEGER_RESET_VALUE);
274 rc &= writeReg(REG_PLL_CTRL, 0x00);
275
276 /* Default clock mode: BCLK number per WCLK = 64 */
277 rc &= writeReg(REG_DAI_CLK_MODE, DAI_BCLKS_PER_WCLK_BCLK64);
278
279 /* Enable DAI, default word length 16 bits, I2S format, output enabled */
280 rc &= writeReg(REG_DAI_CTRL, DAI_EN_MASK | DAI_OE_MASK |
281 DAI_WORD_LENGTH_16B |
282 (uint8_t)DA7212Protocol::I2S);
283
284 /* Route DAC to MixOut by default */
285 rc &= writeReg(REG_DIG_ROUTING_DAC,
286 DIG_ROUTING_DAC_R_RSC_DAC_R | DIG_ROUTING_DAC_L_RSC_DAC_L);
287
288 rc &= setClockMode(master, word_size);
289 rc &= routeDacInput(playback);
290 rc &= setProtocol(protocol);
291 rc &= setAudioFormat(sample_rate, word_size);
292
293 if (playback) rc &= configureOutput();
294 if (capture) rc &= configureInput();
295
296 return rc;
297 }

307 bool setClockMode(bool master, uint8_t word_size) {
308 if (!master) {
309 return updateReg(REG_DAI_CLK_MODE, DAI_CLK_EN_MASK, 0);
310 }
311
312 uint8_t val;
313 switch (word_size) {
314 case 16:
315 val = DAI_BCLKS_PER_WCLK_BCLK32;
316 break;
317 case 32:
318 val = DAI_BCLKS_PER_WCLK_BCLK64;
319 break;
320 case 64:
321 val = DAI_BCLKS_PER_WCLK_BCLK128;
322 break;
323 case 128:
324 val = DAI_BCLKS_PER_WCLK_BCLK256;
325 break;
326 default:
327 val = DAI_BCLKS_PER_WCLK_BCLK32;
328 break;
329 }
330 bool rc = updateReg(REG_DAI_CLK_MODE, DAI_CLK_EN_MASK, DAI_CLK_EN_MASK);
331 rc &= updateReg(REG_DAI_CLK_MODE, DAI_BCLKS_PER_WCLK_MASK, val);
332 return rc;
333 }

536 bool configureOutput() {
537 bool rc = true;
538
539 if (output_device == DAC_OUTPUT_NONE) {
540 /* keep DACs/headphone amps disabled and muted */
541 rc &= setOutputMute(true, DA7212Channel::All);
542 return rc;
543 }
544
545 /* Power charge pump: 0xFD */
546 rc &= writeReg(REG_CP_CTRL, CP_CTRL_EN_MASK |
547 CP_CTRL_SMALL_SWITCH_FREQ_EN_MASK |
548 CP_CTRL_MCHANGE_OUTPUT |
549 CP_CTRL_MOD_CPVDD_1 |
550 CP_CTRL_ANALOG_VLL_LV_BOOSTS_CP);
551
552 rc &= routeOutput();
553
554 /* Enable DACs with ramp */
555 rc &= writeReg(REG_DAC_L_CTRL, EN_MASK | RAMP_EN_MASK);
556 rc &= writeReg(REG_DAC_R_CTRL, EN_MASK | RAMP_EN_MASK);
557
558 /* Enable HP amps, ZC and OE */
559 rc &= writeReg(REG_HP_L_CTRL, AMP_EN_MASK | AMP_RAMP_EN_MASK |
560 AMP_ZC_EN_MASK | AMP_OE_MASK);
561 rc &= writeReg(REG_HP_R_CTRL, AMP_EN_MASK | AMP_RAMP_EN_MASK |
562 AMP_ZC_EN_MASK | AMP_OE_MASK);
563
564 /* Enable MixOut amplifiers and mixing into HP */
565 rc &= writeReg(REG_MIXOUT_L_CTRL, MIXOUT_AMP_EN_MASK |
566 MIXOUT_AMP_SOFT_MIX_EN_MASK |
567 MIXOUT_AMP_MIX_EN_MASK);
568 rc &= writeReg(REG_MIXOUT_R_CTRL, MIXOUT_AMP_EN_MASK |
569 MIXOUT_AMP_SOFT_MIX_EN_MASK |
570 MIXOUT_AMP_MIX_EN_MASK);
571
572 /* DAC gain */
573 rc &= writeReg(REG_DAC_L_GAIN, DAC_DEFAULT_GAIN);
574 rc &= writeReg(REG_DAC_R_GAIN, DAC_DEFAULT_GAIN);
575
576 /* Default HP volume and unmute */
577 rc &= setOutputVolume(HP_DEFAULT_GAIN, DA7212Channel::All);
578 rc &= setOutputMute(false, DA7212Channel::All);
579
580 return rc;
581 }

588 bool configureInput() {
589 bool rc = true;
590
591 rc &= routeInput(DA7212Channel::All);
592
593 /* Charge pump control: 0xFD */
594 rc &= writeReg(REG_CP_CTRL, CP_CTRL_EN_MASK |
595 CP_CTRL_SMALL_SWITCH_FREQ_EN_MASK |
596 CP_CTRL_MCHANGE_OUTPUT |
597 CP_CTRL_MOD_CPVDD_1 |
598 CP_CTRL_ANALOG_VLL_LV_BOOSTS_CP);
599
600 /* AUX_L_CTRL: 0xB4 */
601 rc &= writeReg(REG_AUX_L_CTRL, AUX_AMP_EN_MASK | AUX_AMP_RAMP_EN_MASK |
602 AUX_AMP_ZC_EN_MASK |
603 AUX_L_CTRL_AMP_ZC_SEL_INPUT_AUX_L_IF);
604
605 /* AUX_R_CTRL: 0xB0 */
606 rc &= writeReg(REG_AUX_R_CTRL,
607 AUX_AMP_EN_MASK | AUX_AMP_RAMP_EN_MASK | AUX_AMP_ZC_EN_MASK);
608
609 /* MIC_x_CTRL: 0x04 */
610 rc &= writeReg(REG_MIC_1_CTRL, MIC_CTRL_AMP_IN_SEL_MIC_P);
611 rc &= writeReg(REG_MIC_2_CTRL, MIC_CTRL_AMP_IN_SEL_MIC_P);
612
613 /* MIXIN_L/R_CTRL: 0x88 */
614 rc &= writeReg(REG_MIXIN_L_CTRL, MIXIN_AMP_EN_MASK | MIXIN_AMP_MIX_EN_MASK);
615 rc &= writeReg(REG_MIXIN_R_CTRL, MIXIN_AMP_EN_MASK | MIXIN_AMP_MIX_EN_MASK);
616
617 /* ADC_L/R_CTRL: 0xA0 */
618 rc &= writeReg(REG_ADC_L_CTRL, EN_MASK | RAMP_EN_MASK);
619 rc &= writeReg(REG_ADC_R_CTRL, EN_MASK | RAMP_EN_MASK);
620
621 /* GAIN_RAMP_CTRL: 0x02 */
622 rc &= writeReg(REG_GAIN_RAMP_CTRL, GAIN_RAMP_CTRL_RATE_NR_MUL_16);
623
624 /* PC_COUNT: 0x02 */
625 rc &= writeReg(REG_PC_COUNT, PC_COUNT_RESYNC_MASK);
626
627 /* CP_DELAY: 0x95 */
628 rc &= writeReg(REG_CP_DELAY, CP_DELAY_ON_OFF_LIMITER_AUT |
629 CP_DELAY_TAU_DELAY_4MS |
630 CP_DELAY_FCONTROL_0HZ_OR_1MHZ);
631
632 /* Default ADC volume and unmute */
633 rc &= setInputVolume(HP_DEFAULT_GAIN, DA7212Channel::All);
634 rc &= setInputMute(false, DA7212Channel::All);
635
636 return rc;
637 }
