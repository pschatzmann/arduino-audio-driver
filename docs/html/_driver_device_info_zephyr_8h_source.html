268 void setupPinMode() {
269 gpio.begin(*this);
270 AD_LOGD("DriverDeviceInfo::setupPinMode");
271 // setup pins
272 for (auto& tmp : pins) {
273 if (tmp.pin != -1) {
274 if (!hasConflict(tmp.pin)) {
275 AD_LOGD("pinMode for %d", tmp.pin);
276 switch (tmp.pin_logic) {
277 case PinLogic::InputActiveHigh:
278 gpio.pinMode(tmp.pin, INPUT);
279 break;
280 case PinLogic::InputActiveLow:
281 gpio.pinMode(tmp.pin, INPUT_PULLUP);
282 break;
283 case PinLogic::Input:
284 gpio.pinMode(tmp.pin, INPUT);
285 break;
286 case PinLogic::Output:
287 gpio.pinMode(tmp.pin, OUTPUT);
288 break;
289 default:
290 // do nothing
291 break;
292 }
293 } else {
294 AD_LOGW("Pin '%d' not set up because of conflict", tmp.pin);
295 tmp.active = false;
296 }
297 } else {
298 AD_LOGD("Pin is -1");
299 }
300 }
