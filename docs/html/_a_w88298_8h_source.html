131 bool setFormat(AW88298Mode mode, uint8_t word_size, uint32_t sample_rate) {
132 uint16_t fs_code, bck_code;
133 if (!wordSizeCodes(word_size, fs_code, bck_code)) return false;
134
135 uint16_t rate_code;
136 if (!sampleRateCode(sample_rate, rate_code)) return false;
137
138 bool rc = true;
139 // Enable I2S interface, clear power down
140 rc &= updateReg16(REG_SYSCTRL, SYSCTRL_I2SEN | SYSCTRL_PWDN,
141 SYSCTRL_I2SEN);
142
143 uint16_t mask = I2SCTRL_I2SMD_MASK | I2SCTRL_I2SFS_MASK |
144 I2SCTRL_I2SBCK_MASK | I2SCTRL_I2SSR_MASK;
145 uint16_t value = (((uint16_t)mode << 8) & I2SCTRL_I2SMD_MASK) |
146 ((fs_code << 6) & I2SCTRL_I2SFS_MASK) |
147 ((bck_code << 4) & I2SCTRL_I2SBCK_MASK) |
148 (rate_code & I2SCTRL_I2SSR_MASK);
149 rc &= updateReg16(REG_I2SCTRL, mask, value);
150
151 return rc;
152 }

188 static uint8_t db2vol(int db) {
189 if (db > 0) return 0;
190 int hi = (-db) / 6;
191 if (hi > 0xF) hi = 0xF;
192 uint8_t lo = (uint8_t)(((-db) - (hi * 6)) * 2);
193 return (uint8_t)((hi << 4) | (lo & 0xF));
194 }
