181 bool begin(WM8904Protocol protocol = WM8904Protocol::I2S,
182 uint32_t sample_rate = 44100, uint8_t word_size = 16,
183 bool playback = true, bool capture = true) {
184 bool rc = true;
185
186 rc &= softReset();
187
188 /* MCLK_INV=0, SYSCLK_SRC=0, TOCLK_RATE=0, OPCLK_ENA=1,
189 * CLK_SYS_ENA=1, CLK_DSP_ENA=1, TOCLK_ENA=1 */
190 rc &= writeReg16(REG_CLK_RATES_2, 0x000F);
191
192 /* WSEQ_ENA=1, WSEQ_WRITE_INDEX=0_0000 */
193 rc &= writeReg16(REG_WRT_SEQUENCER_0, 0x0100);
194
195 /* WSEQ_ABORT=0, WSEQ_START=1, WSEQ_START_INDEX=00_0000 */
196 rc &= writeReg16(REG_WRT_SEQUENCER_3, 0x0100);
197
198 // wait until the write sequencer has finished
199 uint16_t value = 1;
200 int retries = 1000;
201 while ((value & 1U) != 0U && retries-- > 0) {
202 if (!readReg16(REG_WRT_SEQUENCER_4, value)) break;
203 }
204
205 /* TOCLK_RATE_DIV16=0, TOCLK_RATE_x4=1, SR_MODE=0, MCLK_DIV=1 */
206 rc &= writeReg16(REG_CLK_RATES_0, 0xA45F);
207
208 /* INL_ENA=1, INR_ENA=1 */
209 rc &= writeReg16(REG_POWER_MGMT_0, 0x0003);
210
211 /* HPL_PGA_ENA=1, HPR_PGA_ENA=1 */
212 rc &= writeReg16(REG_POWER_MGMT_2, 0x0003);
213
214 /* DACL_ENA=1, DACR_ENA=1, ADCL_ENA=1, ADCR_ENA=1 */
215 rc &= writeReg16(REG_POWER_MGMT_6, 0x000F);
216
217 /* ADC_OSR128=1 */
218 rc &= writeReg16(REG_ANALOG_ADC_0, 0x0001);
219
220 /* AIFADCR_SRC=1, AIFDACR_SRC=1 */
221 rc &= writeReg16(REG_AUDIO_IF_0, 0x0050);
222
223 /* DAC_OSR128=1 */
224 rc &= writeReg16(REG_DAC_DIG_1, 0x0040);
225
226 /* Enable DC servos for headphone out */
227 rc &= writeReg16(REG_DC_SERVO_0, 0x0003);
228
229 /* HPL_RMV_SHORT=1, HPL_ENA_OUTP=1, HPL_ENA_DLY=1, HPL_ENA=1,
230 * HPR_RMV_SHORT=1, HPR_ENA_OUTP=1, HPR_ENA_DLY=1, HPR_ENA=1 */
231 rc &= writeReg16(REG_ANALOG_HP_0, 0x00FF);
232
233 /* CP_DYN_PWR=1 */
234 rc &= writeReg16(REG_CLS_W_0, 0x0001);
235
236 /* CP_ENA=1 */
237 rc &= writeReg16(REG_CHRG_PUMP_0, 0x0001);
238
239 rc &= setProtocol(protocol);
240 rc &= setAudioFormat(sample_rate, word_size);
241
242 if (playback) rc &= configureOutput();
243 if (capture) rc &= configureInput();
244
245 return rc;
246 }

266 bool setAudioFormat(uint32_t sample_rate, uint8_t word_size = 16) {
267 bool rc = true;
268 WM8904SampleRate wm_sample_rate;
269
270 switch (sample_rate) {
271 case 8000:
272 wm_sample_rate = WM8904SampleRate::Rate8kHz;
273 break;
274 case 11025:
275 case 12000:
276 wm_sample_rate = WM8904SampleRate::Rate12kHz;
277 break;
278 case 16000:
279 wm_sample_rate = WM8904SampleRate::Rate16kHz;
280 break;
281 case 22050:
282 case 24000:
283 wm_sample_rate = WM8904SampleRate::Rate24kHz;
284 break;
285 case 32000:
286 wm_sample_rate = WM8904SampleRate::Rate32kHz;
287 break;
288 case 44100:
289 case 48000:
290 wm_sample_rate = WM8904SampleRate::Rate48kHz;
291 break;
292 default:
293 return false;
294 }
295
296 /* Disable SYSCLK */
297 rc &= writeReg16(REG_CLK_RATES_2, 0x00);
298
299 /* Set Clock ratio and sample rate (fs ratio fixed to 256X, matches
300 * MCLK_DIV=1 in REG_CLK_RATES_0 written during begin()) */
301 rc &= writeReg16(REG_CLK_RATES_1, ((uint16_t)WM8904FsRatio::Ratio256X
302 << 10U) |
303 (uint16_t)wm_sample_rate);
304
305 uint16_t word_size_bits = 0;
306 switch (word_size) {
307 case 16:
308 word_size_bits = 0;
309 break;
310 case 20:
311 word_size_bits = 1;
312 break;
313 case 24:
314 word_size_bits = 2;
315 break;
316 case 32:
317 word_size_bits = 3;
318 break;
319 default:
320 word_size_bits = 0;
321 break;
322 }
323 /* Set bit resolution */
324 rc &= updateReg16(REG_AUDIO_IF_1, 0x000CU, (word_size_bits << 2U));
325
326 /* Enable SYSCLK */
327 rc &= writeReg16(REG_CLK_RATES_2, 0x1007);
328
329 return rc;
330 }

404 bool routeInput(uint32_t input, WM8904Channel channel = WM8904Channel::All) {
405 if (input < 1 || input > 3) return false;
406
407 uint8_t val = (uint8_t)(((input - 1) & 0x03) << 4U) |
408 (uint8_t)(((input - 1) & 0x03) << 2U);
409 uint8_t mask = REGMASK_INSEL_CMENA | REGMASK_INSEL_IP_SEL_P |
410 REGMASK_INSEL_IP_SEL_N | REGMASK_INSEL_MODE;
411
412 bool rc = true;
413 if (channel == WM8904Channel::FrontLeft || channel == WM8904Channel::All)
414 rc &= updateReg(REG_ANALOG_LEFT_IN_1, mask, val);
415 if (channel == WM8904Channel::FrontRight || channel == WM8904Channel::All)
416 rc &= updateReg(REG_ANALOG_RIGHT_IN_1, mask, val);
417 return rc;
418 }

433 bool configureOutput() {
434 bool rc = true;
435 bool hp = output_device == DAC_OUTPUT_LINE1 || output_device == DAC_OUTPUT_ALL;
436 bool spk = output_device == DAC_OUTPUT_LINE2 || output_device == DAC_OUTPUT_ALL;
437
438 rc &= setOutputVolume((uint8_t)OUTPUT_VOLUME_DEFAULT, WM8904Channel::All);
439 rc &= setOutputMute(!hp, WM8904Channel::HeadphoneLeft);
440 rc &= setOutputMute(!hp, WM8904Channel::HeadphoneRight);
441 rc &= setOutputMute(!spk, WM8904Channel::FrontLeft);
442 rc &= setOutputMute(!spk, WM8904Channel::FrontRight);
443 rc &= applyProperties();
444 return rc;

461 bool updateOutput(WM8904Channel channel, uint16_t val, uint16_t mask) {
462 bool rc = true;
463 switch (channel) {
464 case WM8904Channel::FrontLeft:
465 return updateReg16(REG_ANALOG_OUT2_LEFT, mask, val);
466 case WM8904Channel::FrontRight:
467 return updateReg16(REG_ANALOG_OUT2_RIGHT, mask, val);
468 case WM8904Channel::HeadphoneLeft:
469 return updateReg16(REG_ANALOG_OUT1_LEFT, mask, val);
470 case WM8904Channel::HeadphoneRight:
471 return updateReg16(REG_ANALOG_OUT1_RIGHT, mask, val);
472 case WM8904Channel::All:
473 rc &= updateReg16(REG_ANALOG_OUT1_LEFT, mask, val);
474 rc &= updateReg16(REG_ANALOG_OUT1_RIGHT, mask, val);
475 rc &= updateReg16(REG_ANALOG_OUT2_LEFT, mask, val);
476 rc &= updateReg16(REG_ANALOG_OUT2_RIGHT, mask, val);
477 return rc;
478 default:
479 return false;
480 }
481 }

492 rc &= updateReg(REG_ANALOG_RIGHT_IN_0, mask, val);
