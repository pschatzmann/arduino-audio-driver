166 uint32_t init(uint32_t inputDevice, uint32_t outputDevice, uint8_t volume,
167 uint32_t audioFreq) {
168 uint32_t ret = 0;
169 uint8_t tmp;
170
171 if (is_initialized == 0) {
172 /* Set the device in standby mode */
173 tmp = readReg(CS42L51_PWR_CTRL_1);
174 tmp |= 0x01U;
175 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
176
177 /* Set all power down bits to 1 */
178 tmp = 0x7FU;
179 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
180 tmp = readReg(CS42L51_MIC_PWR_SPEED_CTRL);
181 tmp |= 0x0EU;
182 ret += writeReg(CS42L51_MIC_PWR_SPEED_CTRL, tmp);
183
184 is_initialized = 1;
185 } else {
186 /* Set all power down bits to 1 except PDN to mute ADCs and DACs*/
187 tmp = 0x7EU;
188 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
189 tmp = readReg(CS42L51_MIC_PWR_SPEED_CTRL);
190 tmp |= 0x0EU;
191 ret += writeReg(CS42L51_MIC_PWR_SPEED_CTRL, tmp);
192
193 /* Disable zero cross and soft ramp */
194 tmp = readReg(CS42L51_DAC_CTRL);
195 tmp &= 0xFCU;
196 ret += writeReg(CS42L51_DAC_CTRL, tmp);
197
198 /* Power control : Enter standby (PDN = 1) */
199 tmp = readReg(CS42L51_PWR_CTRL_1);
200 tmp |= 0x01U;
201 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
202 }
203
204 /* Mic Power and Speed Control : Auto detect on, Speed mode SSM, tri state
205 * off, MCLK divide by 2 off */
206 tmp = readReg(CS42L51_MIC_PWR_SPEED_CTRL);
207 tmp = ((tmp & 0x0EU) | 0xA0U);
208 ret += writeReg(CS42L51_MIC_PWR_SPEED_CTRL, tmp);
209
210 /* Interface control : Loopback off, Slave, I2S (SDIN and SOUT), Digital
211 * mix off, Mic mix off */
212 tmp = 0x0CU;
213 ret += writeReg(CS42L51_INTERFACE_CTRL, tmp);
214
215 /* Mic control : ADC single volume off, ADCB boost off, ADCA boost off,
216 * MicBias on AIN3B/MICIN2 pin, MicBias level 0.8xVA, MICB boost 32db,
217 * MICA boost 32dB */
218 tmp = 0x03U;
219 ret += writeReg(CS42L51_MIC_CTRL, tmp);
220
221 /* ADC control : ADCB HPF on, ADCB HPF freeze off, ADCA HPF on, ADCA HPF
222 * freeze off, Soft ramp B on, Zero cross B on, Soft ramp A on, Zero cross
223 * A on */
224 tmp = 0xAFU;
225 ret += writeReg(CS42L51_ADC_CTRL, tmp);
226
227 /* DAC output control : HP Gain to 1, Single volume control off, PCM
228 * invert signals polarity off, DAC channels mute on */
229 tmp = 0xC3U;
230 ret += writeReg(CS42L51_DAC_OUTPUT_CTRL, tmp);
231
232 /* DAC control : Signal processing to DAC, Freeze off, De-emphasis off,
233 * Analog output auto mute off, DAC soft ramp */
234 tmp = 0x42U;
235 ret += writeReg(CS42L51_DAC_CTRL, tmp);
236
237 /* ALCA and PGAA Control : ALCA soft ramp disable on, ALCA zero cross
238 * disable on, PGA A Gain +8dB */
239 tmp = 0xD0U;
240 ret += writeReg(CS42L51_ALCA_PGAA_CTRL, tmp);
241
242 /* ALCB and PGAB Control : ALCB soft ramp disable on, ALCB zero cross
243 * disable on, PGA B Gain +8dB */
244 ret += writeReg(CS42L51_ALCB_PGAB_CTRL, tmp);
245
246 if ((inputDevice & CS42L51_IN_LINE1) == CS42L51_IN_LINE1) {
247 /* ADCA/ADCB Attenuator : -10dB */
248 tmp = 0xF6U;
249 } else {
250 /* ADCA/ADCB Attenuator : 0dB */
251 tmp = 0x00U;
252 }
253 /* ADCA Attenuator */
254 ret += writeReg(CS42L51_ADCA_ATTENUATOR, tmp);
255 /* ADCB Attenuator */
256 ret += writeReg(CS42L51_ADCB_ATTENUATOR, tmp);
257
258 /* ADCA mixer volume control : ADCA mixer channel mute on, ADCA mixer
259 * volume 0dB */
260 tmp = 0x80U;
261 ret += writeReg(CS42L51_ADCA_MIXER_VOL_CTRL, tmp);
262 /* ADCB mixer volume control : ADCB mixer channel mute on, ADCB mixer
263 * volume 0dB */
264 ret += writeReg(CS42L51_ADCB_MIXER_VOL_CTRL, tmp);
265
266 /* PCMA mixer volume control : PCMA mixer channel mute off, PCMA mixer
267 * volume 0dB */
268 tmp = 0x00U;
269 ret += writeReg(CS42L51_PCMA_MIXER_VOL_CTRL, tmp);
270 /* PCMB mixer volume control : PCMB mixer channel mute off, PCMB mixer
271 * volume 0dB */
272 ret += writeReg(CS42L51_PCMB_MIXER_VOL_CTRL, tmp);
273 /* PCM channel mixer : AOUTA Left, AOUTB Right */
274 ret += writeReg(CS42L51_ADC_PCM_CHANNEL_MIXER, tmp);
275
276 if ((outputDevice & CS42L51_OUT_HEADPHONE) == CS42L51_OUT_HEADPHONE) {
277 tmp = volumeConvert(volume);
278 /* AOUTA volume control : AOUTA volume */
279 ret += writeReg(CS42L51_AOUTA_VOL_CTRL, tmp);
280 /* AOUTB volume control : AOUTB volume */
281 ret += writeReg(CS42L51_AOUTB_VOL_CTRL, tmp);
282 }
283
284 /* ALC enable and attack rate : ALCB and ALCA enable, fastest attack */
285 tmp = 0x40U;
286 ret += writeReg(CS42L51_ALC_ENABLE_AND_ATTACK_RATE, tmp);
287
288 /* Store current devices */
289 current_devices = (outputDevice | inputDevice);
290
291 return ret;
292 }

304 uint32_t play() {
305 uint32_t ret = 0;
306 uint8_t tmp;
307
308 if ((current_devices & CS42L51_OUT_HEADPHONE) == CS42L51_OUT_HEADPHONE) {
309 /* Unmute the output first */
310 ret += setMute(CS42L51_MUTE_OFF);
311
312 /* DAC control : Signal processing to DAC, Freeze off, De-emphasis off,
313 * Analog output auto mute off, DAC soft ramp */
314 tmp = 0x42U;
315 ret += writeReg(CS42L51_DAC_CTRL, tmp);
316
317 /* Power control 1 : PDN_DACA, PDN_DACB disable. */
318 tmp = readReg(CS42L51_PWR_CTRL_1);
319 tmp &= 0x9FU;
320 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
321 }
322
323 if ((current_devices & CS42L51_IN_LINE1) == CS42L51_IN_LINE1) {
324 /* ADC Input Select, Invert and Mute : AIN1B to PGAB, AIN1A to PGAA,
325 * ADCB invert off, ADCA invert off, ADCB mute off, ADCA mute off */
326 tmp = 0x00U;
327 ret += writeReg(CS42L51_ADCX_INPUT_SELECT, tmp);
328
329 /* Power control 1 : PDN_PGAA, PDN_PGAA, PDN_ADCB, PDN_ADCA disable. */
330 tmp = readReg(CS42L51_PWR_CTRL_1);
331 tmp &= 0xF9U;
332 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
333
334 /* Mic Power and Speed Control : PDN_MICA, PDN_MICB, PDN_MIC_BIAS
335 * disable. */
336 tmp = readReg(CS42L51_MIC_PWR_SPEED_CTRL);
337 tmp &= 0xFFU;
338 ret += writeReg(CS42L51_MIC_PWR_SPEED_CTRL, tmp);
339 }
340
341 if ((current_devices & CS42L51_IN_MIC1) == CS42L51_IN_MIC1) {
342 /* ADC Input Select, Invert and Mute : AIN1B to PGAB, AIN3A to PreAmp to
343 * PGAA, ADCB invert off, ADCA invert off, ADCB mute on, ADCA mute off */
344 tmp = 0x32U;
345 ret += writeReg(CS42L51_ADCX_INPUT_SELECT, tmp);
346
347 /* Power control 1 : PDN_PGAA, PDN_ADCA disable. */
348 tmp = readReg(CS42L51_PWR_CTRL_1);
349 tmp &= 0xF5U;
350 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
351
352 /* Mic Power and Speed Control : PDN_MICA, PDN_MIC_BIAS disable. */
353 tmp = readReg(CS42L51_MIC_PWR_SPEED_CTRL);
354 tmp &= 0xF9U;
355 ret += writeReg(CS42L51_MIC_PWR_SPEED_CTRL, tmp);
356 }
357
358 if ((current_devices & CS42L51_IN_MIC2) == CS42L51_IN_MIC2) {
359 /* Power control 1 : PDN_PGAB, PDN_ADCB disable. */
360 tmp = readReg(CS42L51_PWR_CTRL_1);
361 tmp &= 0xEBU;
362 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
363
364 /* Mic Power and Speed Control : PDN_MICB, PDN_MIC_BIAS disable. */
365 tmp = readReg(CS42L51_MIC_PWR_SPEED_CTRL);
366 tmp &= 0xF5U;
367 ret += writeReg(CS42L51_MIC_PWR_SPEED_CTRL, tmp);
368 }
369
370 /* Power control : Exit standby (PDN = 0) */
371 tmp = readReg(CS42L51_PWR_CTRL_1);
372 tmp &= 0xFEU;
373 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
374
375 return ret;
376 }

387 uint32_t stop(uint32_t cmd) {
388 uint32_t ret = 0;
389 uint8_t tmp;
390
391 /* Mute the output first */
392 ret += setMute(CS42L51_MUTE_ON);
393
394 if (cmd == CS42L51_PDWN_SW) {
395 /* Only output mute required*/
396 } else /* CS42L51_PDWN_HW */
397 {
398 /* Set all power down bits to 1 except PDN to mute ADCs and DACs*/
399 tmp = 0x7EU;
400 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
401 tmp = readReg(CS42L51_MIC_PWR_SPEED_CTRL);
402 tmp |= 0x0EU;
403 ret += writeReg(CS42L51_MIC_PWR_SPEED_CTRL, tmp);
404
405 /* Disable zero cross and soft ramp */
406 tmp = readReg(CS42L51_DAC_CTRL);
407 tmp &= 0xFCU;
408 ret += writeReg(CS42L51_DAC_CTRL, tmp);
409
410 /* Power control : Enter standby (PDN = 1) */
411 tmp = readReg(CS42L51_PWR_CTRL_1);
412 tmp |= 0x01U;
413 ret += writeReg(CS42L51_PWR_CTRL_1, tmp);
414 }

446 uint32_t setMute(uint32_t cmd) {
447 uint32_t ret = 0;
448 uint8_t tmp = readReg(CS42L51_DAC_OUTPUT_CTRL);
449
450 /* Set the Mute mode */
451 if (cmd == CS42L51_MUTE_ON) {

457

459

485 inline constexpr uint8_t volumeConvert(uint8_t volume) {
486 return volume >= 100 ? static_cast<uint8_t>(0)
