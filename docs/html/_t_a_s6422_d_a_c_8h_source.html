159 bool softReset() {
160 uint8_t val = 0;
161 if (!readReg(MODE_CTRL_ADDR, val)) return false;
162 val |= MODE_CTRL_RESET;
163 return writeReg(MODE_CTRL_ADDR, val);

172 bool configureDai(uint32_t sample_rate) {
173 uint8_t val = 0;
174 if (!readReg(SAP_CTRL_ADDR, val)) return false;
175
176 val &= (uint8_t)~SAP_CTRL_INPUT_FORMAT_MASK;
177 val |= (uint8_t)TAS6422InputFormat::I2S;
178
179 val &= (uint8_t)~SAP_CTRL_INPUT_SAMPLING_RATE_MASK;
180 switch (sample_rate) {
181 case 44100:
182 val |= (SAP_CTRL_INPUT_SAMPLING_RATE_44_1_KHZ << 6);
183 break;
184 case 48000:
185 val |= (SAP_CTRL_INPUT_SAMPLING_RATE_48_KHZ << 6);
186 break;
187 case 96000:
188 val |= (SAP_CTRL_INPUT_SAMPLING_RATE_96_KHZ << 6);
189 break;
190 default:
191 return false;
192 }
193
194 return writeReg(SAP_CTRL_ADDR, val);
195 }
196
207 bool configureOutput() {
208 bool rc = true;
209 uint8_t val = 0;
210
211 if (output_device == DAC_OUTPUT_NONE || output_device == DAC_OUTPUT_LINE1) {
212 return setChannelState(TAS6422Channel::All, TAS6422ChannelState::Mute);
213 }
214
215 rc &= readReg(MISC_CTRL_1_ADDR, val);
216 val &= (uint8_t)~MISC_CTRL_1_OC_CONTROL_MASK;
217 rc &= writeReg(MISC_CTRL_1_ADDR, val);
218
219 rc &= readReg(MISC_CTRL_2_ADDR, val);
220 val &= (uint8_t)~MISC_CTRL_2_PWM_FREQUENCY_MASK;
221 val |= (uint8_t)(MISC_CTRL_2_PWM_FREQUENCY_10_FS << 4);
222 rc &= writeReg(MISC_CTRL_2_ADDR, val);
223
224 return rc;
225 }

228 bool setChannelState(TAS6422Channel channel, TAS6422ChannelState state) {
229 uint8_t val = 0;
230 if (!readReg(CH_STATE_CTRL_ADDR, val)) return false;
231
232 switch (channel) {
233 case TAS6422Channel::Channel1:
234 val &= (uint8_t)~CH_STATE_CTRL_CH1_STATE_CTRL_MASK;
235 val |= (uint8_t)((uint8_t)state << 6);
236 break;
237 case TAS6422Channel::Channel2:
238 val &= (uint8_t)~CH_STATE_CTRL_CH2_STATE_CTRL_MASK;
239 val |= (uint8_t)((uint8_t)state << 4);
240 break;
241 case TAS6422Channel::All:
242 val &= (uint8_t)~(CH_STATE_CTRL_CH1_STATE_CTRL_MASK |
243 CH_STATE_CTRL_CH2_STATE_CTRL_MASK);
244 val |= (uint8_t)((uint8_t)state << 6) | (uint8_t)((uint8_t)state << 4);
245 break;
246 default:
247 return false;
248 }
249
250 return writeReg(CH_STATE_CTRL_ADDR, val);
251 }

283 bool setOutputVolume(int volume, TAS6422Channel channel = TAS6422Channel::All) {
284 if (volume > OUTPUT_VOLUME_MAX || volume < OUTPUT_VOLUME_MIN) return false;
285
286 uint8_t vol_val = (uint8_t)(volume + 0xCF);
287
288 switch (channel) {
289 case TAS6422Channel::Channel1:
290 return writeReg(CH1_VOLUME_CTRL_ADDR, vol_val);
291 case TAS6422Channel::Channel2:
292 return writeReg(CH2_VOLUME_CTRL_ADDR, vol_val);
293 case TAS6422Channel::All: {
294 bool rc = true;
295 rc &= writeReg(CH1_VOLUME_CTRL_ADDR, vol_val);
296 rc &= writeReg(CH2_VOLUME_CTRL_ADDR, vol_val);
297 return rc;
298 }
299 default:
300 return false;
301 }

310 return setOutputVolume(vol, TAS6422Channel::All);
