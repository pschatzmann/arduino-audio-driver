221 bool begin(WM8962Protocol protocol = WM8962Protocol::I2S,
222 uint32_t sample_rate = 44100, uint8_t word_size = 16,
223 uint32_t mclk_to_fs_ratio = 256, bool playback = true,

572 bool runSequence(WM8962Sequence id) {
573 uint32_t delay_remaining_ms;
574 switch (id) {
575 case WM8962Sequence::DACToHeadphonePowerUp:
576 delay_remaining_ms = 93;
577 break;
578 case WM8962Sequence::AnalogueInputPowerUp:
579 delay_remaining_ms = 75;
580 break;
581 case WM8962Sequence::ChipPowerDown:
582 delay_remaining_ms = 32;
583 break;
584 case WM8962Sequence::SpeakerSleep:
585 case WM8962Sequence::SpeakerWake:
586 default:
587 delay_remaining_ms = 2;
588 break;
589 }
590
591 bool rc = true;
592 rc &= writeRegWide(REG_WRITE_SEQ_CTRL_1, WSEQ_ENA);
593 rc &= writeRegWide(REG_WRITE_SEQ_CTRL_2, (uint16_t)id);
594
595 uint16_t status = 1;
596 while (delay_remaining_ms > 0) {
597 if (!readRegWide(REG_WRITE_SEQ_CTRL_3, status)) break;
598 if ((status & 1U) == 0U) break;
599 delayMs(1);
600 delay_remaining_ms--;
601 }
602 return rc && (status & 1U) == 0U;
603 }

611 bool writeRegWide(uint16_t reg, uint16_t value) {
612 assert(wire != nullptr);
613 uint8_t data[4] = {
614 (uint8_t)(reg >> 8), (uint8_t)(reg & 0xFF),
615 (uint8_t)(value >> 8), (uint8_t)(value & 0xFF),
616 };
617 return i2c_bus_write_bytes(wire, i2c_addr, data, 2, data + 2, 2) == 0;
618 }
619
622 bool readRegWide(uint16_t reg, uint16_t& value) {
623 assert(wire != nullptr);
624 uint8_t reg_bytes[2] = {(uint8_t)(reg >> 8), (uint8_t)(reg & 0xFF)};
625 uint8_t data[2] = {0, 0};
626 if (i2c_bus_read_bytes(wire, i2c_addr, reg_bytes, 2, data, 2) != 0)
627 return false;
628 value = ((uint16_t)data[0] << 8) | data[1];
629 return true;
630 }
631
633 bool updateRegWide(uint16_t reg, uint16_t mask, uint16_t value) {
634 uint16_t old = 0;
635 if (!readRegWide(reg, old)) return false;
636 uint16_t updated = (old & ~mask) | (value & mask);
637 if (updated == old) return true;
638 return writeRegWide(reg, updated);
639 }

641 bool updateOutput(WM8962Channel channel, uint16_t val, uint16_t mask) {
642 bool rc = true;
643 switch (channel) {
644 case WM8962Channel::FrontLeft:
645 return updateRegWide(REG_LOUT2, mask, val);
646 case WM8962Channel::FrontRight:
647 return updateRegWide(REG_ROUT2, mask, val);
648 case WM8962Channel::HeadphoneLeft:
649 return updateRegWide(REG_LOUT1, mask, val);
650 case WM8962Channel::HeadphoneRight:
651 return updateRegWide(REG_ROUT1, mask, val);
652 case WM8962Channel::All:
653 rc &= updateRegWide(REG_LOUT1, mask, val);
654 rc &= updateRegWide(REG_ROUT1, mask, val);
655 rc &= updateRegWide(REG_LOUT2, mask, val);
656 rc &= updateRegWide(REG_ROUT2, mask, val);
657 return rc;
658 default:
659 return false;
660 }
