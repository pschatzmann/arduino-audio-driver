55struct es8311_coeff_div {
56 uint32_t mclk; /* mclk frequency */
57 uint32_t rate; /* sample rate */
58 uint8_t pre_div; /* the pre divider with range from 1 to 8 */
59 uint8_t pre_multi; /* the pre multiplier with x1, x2, x4 and x8 selection */
60 uint8_t adc_div; /* adcclk divider */
61 uint8_t dac_div; /* dacclk divider */
62 uint8_t fs_mode; /* double speed or single speed, =0, ss, =1, ds */
63 uint8_t lrck_h; /* adclrck divider and daclrck divider */
64 uint8_t lrck_l;
65 uint8_t bclk_div; /* sclk divider */
66 uint8_t adc_osr; /* adc osr */
67 uint8_t dac_osr; /* dac osr */
68};

313 error_t init(codec_config_t* codec_cfg) {
314 uint8_t datmp, regv;
315 int coeff;
316 error_t ret = RESULT_OK;
317 assert(i2c_handle != NULL);
318
319 ret |= writeReg(ES8311_CLK_MANAGER_REG01, 0x30);
320 ret |= writeReg(ES8311_CLK_MANAGER_REG02, 0x00);
321 ret |= writeReg(ES8311_CLK_MANAGER_REG03, 0x10);
322 ret |= writeReg(ES8311_ADC_REG16, 0x24);
323 ret |= writeReg(ES8311_CLK_MANAGER_REG04, 0x10);
324 ret |= writeReg(ES8311_CLK_MANAGER_REG05, 0x00);
325 ret |= writeReg(ES8311_SYSTEM_REG0B, 0x00);
326 ret |= writeReg(ES8311_SYSTEM_REG0C, 0x00);
327 ret |= writeReg(ES8311_SYSTEM_REG10, 0x1F);
328 ret |= writeReg(ES8311_SYSTEM_REG11, 0x7F);
329 ret |= writeReg(ES8311_RESET_REG00, 0x80);
330 /*
331 * Set Codec into Master or Slave mode
332 */
333 regv = readReg(ES8311_RESET_REG00);
334 /*
335 * Set master/slave audio interface
336 */
337 I2SDefinition* i2s_cfg = &(codec_cfg->i2s);
338 switch (i2s_cfg->mode) {
339 case MODE_MASTER: /* MASTER MODE */
340 AD_LOGI("ES8311 in Master mode");
341 regv |= 0x40;
342 break;
343 case MODE_SLAVE: /* SLAVE MODE */
344 AD_LOGI("ES8311 in Slave mode");
345 regv &= 0xBF;
346 break;
347 default:
348 regv &= 0xBF;
349 }
350 ret |= writeReg(ES8311_RESET_REG00, regv);
351 ret |= writeReg(ES8311_CLK_MANAGER_REG01, 0x3F);
352 /*
353 * Select clock source for internal mclk
354 */
355 switch (getMclkSrc()) {
356 case FROM_MCLK_PIN:
357 AD_LOGI("ES8311 clock source: MCLK");
358 regv = readReg(ES8311_CLK_MANAGER_REG01);
359 regv &= 0x7F;
360 ret |= writeReg(ES8311_CLK_MANAGER_REG01, regv);
361 break;
362 case FROM_SCLK_PIN:
363 AD_LOGI("ES8311 clock source: SCLK");
364 regv = readReg(ES8311_CLK_MANAGER_REG01);
365 regv |= 0x80;
366 ret |= writeReg(ES8311_CLK_MANAGER_REG01, regv);
367 break;
368 default:
369 AD_LOGI("ES8311 clock source: MCLK");
370 regv = readReg(ES8311_CLK_MANAGER_REG01);
371 regv &= 0x7F;
372 ret |= writeReg(ES8311_CLK_MANAGER_REG01, regv);
373 break;
374 }
375 int sample_fre = 0;
376 int mclk_fre = 0;
377 switch (i2s_cfg->rate) {
378 case RATE_8K:
379 sample_fre = 8000;
380 break;
381 case RATE_11K:
382 sample_fre = 11025;
383 break;
384 case RATE_16K:
385 sample_fre = 16000;
386 break;
387 case RATE_22K:
388 sample_fre = 22050;
389 break;
390 case RATE_24K:
391 sample_fre = 24000;
392 break;
393 case RATE_32K:
394 sample_fre = 32000;
395 break;
396 case RATE_44K:
397 sample_fre = 44100;
398 break;
399 case RATE_48K:
400 sample_fre = 48000;
401 break;
402 default:
403 AD_LOGE("Unable to configure sample rate %dHz", sample_fre);
404 break;
405 }
406 mclk_fre = sample_fre * MCLK_DIV_FRE;
407 coeff = getCoeff(mclk_fre, sample_fre);
408 if (coeff < 0) {
409 AD_LOGE("Unable to configure sample rate %dHz with %dHz MCLK", sample_fre,
410 mclk_fre);
411 return RESULT_FAIL;
412 }
413 /*
414 * Set clock parammeters
415 */
416 if (coeff >= 0) {
417 regv = readReg(ES8311_CLK_MANAGER_REG02) & 0x07;
418 regv |= (es8311_coeff_div[coeff].pre_div - 1) << 5;
419 datmp = 0;
420 switch (es8311_coeff_div[coeff].pre_multi) {
421 case 1:
422 datmp = 0;
423 break;
424 case 2:
425 datmp = 1;
426 break;
427 case 4:
428 datmp = 2;
429 break;
430 case 8:
431 datmp = 3;
432 break;
433 default:
434 break;
435 }
436
437 if (getMclkSrc() == FROM_SCLK_PIN) {
438 datmp = 3; /* DIG_MCLK = LRCK * 256 = BCLK * 8 */
439 }
440 regv |= (datmp) << 3;
441 ret |= writeReg(ES8311_CLK_MANAGER_REG02, regv);
442
443 regv = readReg(ES8311_CLK_MANAGER_REG05) & 0x00;
444 regv |= (es8311_coeff_div[coeff].adc_div - 1) << 4;
445 regv |= (es8311_coeff_div[coeff].dac_div - 1) << 0;
446 ret |= writeReg(ES8311_CLK_MANAGER_REG05, regv);
447
448 regv = readReg(ES8311_CLK_MANAGER_REG03) & 0x80;
449 regv |= es8311_coeff_div[coeff].fs_mode << 6;
450 regv |= es8311_coeff_div[coeff].adc_osr << 0;
451 ret |= writeReg(ES8311_CLK_MANAGER_REG03, regv);
452
453 regv = readReg(ES8311_CLK_MANAGER_REG04) & 0x80;
454 regv |= es8311_coeff_div[coeff].dac_osr << 0;
455 ret |= writeReg(ES8311_CLK_MANAGER_REG04, regv);
456
457 regv = readReg(ES8311_CLK_MANAGER_REG07) & 0xC0;
458 regv |= es8311_coeff_div[coeff].lrck_h << 0;
459 ret |= writeReg(ES8311_CLK_MANAGER_REG07, regv);
460
461 regv = readReg(ES8311_CLK_MANAGER_REG08) & 0x00;
462 regv |= es8311_coeff_div[coeff].lrck_l << 0;
463 ret |= writeReg(ES8311_CLK_MANAGER_REG08, regv);
464
465 regv = readReg(ES8311_CLK_MANAGER_REG06) & 0xE0;
466 if (es8311_coeff_div[coeff].bclk_div < 19) {
467 regv |= (es8311_coeff_div[coeff].bclk_div - 1) << 0;
468 } else {
469 regv |= (es8311_coeff_div[coeff].bclk_div) << 0;
470 }
471 ret |= writeReg(ES8311_CLK_MANAGER_REG06, regv);
472 }
473
474 /*
475 * mclk inverted or not
476 */
477 if (INVERT_MCLK) {
478 regv = readReg(ES8311_CLK_MANAGER_REG01);
479 regv |= 0x40;
480 ret |= writeReg(ES8311_CLK_MANAGER_REG01, regv);
481 } else {
482 regv = readReg(ES8311_CLK_MANAGER_REG01);
483 regv &= ~(0x40);
484 ret |= writeReg(ES8311_CLK_MANAGER_REG01, regv);
485 }
486 /*
487 * sclk inverted or not
488 */
489 if (INVERT_SCLK) {
490 regv = readReg(ES8311_CLK_MANAGER_REG06);
491 regv |= 0x20;
492 ret |= writeReg(ES8311_CLK_MANAGER_REG06, regv);
493 } else {
494 regv = readReg(ES8311_CLK_MANAGER_REG06);
495 regv &= ~(0x20);
496 ret |= writeReg(ES8311_CLK_MANAGER_REG06, regv);
497 }
498
499 ret |= writeReg(ES8311_SYSTEM_REG13, 0x10);
500 ret |= writeReg(ES8311_ADC_REG1B, 0x0A);
501 ret |= writeReg(ES8311_ADC_REG1C, 0x6A);
502
503 // setPaPower(true);
504 return RESULT_OK;
505 }

520 error_t ret = RESULT_OK;
521 codec_mode_t es_mode = CODEC_MODE_MIN;
522

527 case CODEC_MODE_LINE_IN:

531 es_mode = CODEC_MODE_DECODE;
532 break;

544 } else {

553 error_t setVoiceVolume(int volume) {
554 AD_LOGD("setVoiceVolume:%d", volume);
555 if (i2c_handle == 0) return RESULT_FAIL;
556 error_t res = RESULT_OK;
557 if (volume < 0) {
558 volume = 0;
559 } else if (volume > 100) {
560 volume = 100;
561 }
562 // Use a logarithmic scale for more natural volume control
563 int vol = 0;
564 if (volume == 0) {
565 vol = 0;
566 } else {
567 vol = (int)(255.0 * log10(9.0 * volume / 100.0 + 1.0) / log10(10.0));
568 }
569 writeReg(ES8311_DAC_REG32, vol);
570 return res;
571 }
572
574 error_t getVoiceVolume(int* volume) {
575 if (i2c_handle == 0) return RESULT_FAIL;
576 error_t res = RESULT_OK;
577 int regv = 0;
578 regv = readReg(ES8311_DAC_REG32);
579 if (regv == RESULT_FAIL) {
580 *volume = 0;
581 res = RESULT_FAIL;
582 } else {
583 // Inverse of the logarithmic mapping used in setVoiceVolume
584 if (regv <= 0) {
585 *volume = 0;
586 } else {
587 double v = (double)regv / 255.0;
588 *volume = (int)((pow(10.0, v * log10(10.0)) - 1.0) / 9.0 * 100.0 + 0.5);
589 if (*volume > 100) *volume = 100;
590 }
591 }
592 AD_LOGD("GET: res:%d, volume:%d", regv, *volume);
593 return res;
594 }

751 error_t start(codec_mode_t mode) {
752 error_t ret = RESULT_OK;
753 uint8_t adc_iface = 0, dac_iface = 0;
754
755 dac_iface = readReg(ES8311_SDPIN_REG09) & 0xBF;
756 adc_iface = readReg(ES8311_SDPOUT_REG0A) & 0xBF;
757 adc_iface |= bitVal(6);
758 dac_iface |= bitVal(6);
759
760 if (mode == CODEC_MODE_LINE_IN) {
761 AD_LOGE("The codec es8311 doesn't support CODEC_MODE_LINE_IN mode");
762 return RESULT_FAIL;
763 }
764 if (mode == CODEC_MODE_ENCODE || mode == CODEC_MODE_BOTH) {
765 adc_iface &= ~(bitVal(6));
766 }
767 if (mode == CODEC_MODE_DECODE || mode == CODEC_MODE_BOTH) {
768 dac_iface &= ~(bitVal(6));
769 }
770
771 ret |= writeReg(ES8311_SDPIN_REG09, dac_iface);
772 ret |= writeReg(ES8311_SDPOUT_REG0A, adc_iface);
773
774 ret |= writeReg(ES8311_ADC_REG17, 0xBF);
775 ret |= writeReg(ES8311_SYSTEM_REG0E, 0x02);
776 ret |= writeReg(ES8311_SYSTEM_REG12, 0x00);
777 ret |= writeReg(ES8311_SYSTEM_REG14, 0x1A);
778
779 /*
780 * pdm dmic enable or disable
781 */
782 int regv = 0;
783 if (IS_DMIC) {
784 regv = readReg(ES8311_SYSTEM_REG14);
785 regv |= 0x40;
786 ret |= writeReg(ES8311_SYSTEM_REG14, regv);
787 } else {
788 regv = readReg(ES8311_SYSTEM_REG14);
789 regv &= ~(0x40);
790 ret |= writeReg(ES8311_SYSTEM_REG14, regv);
791 }
792
793 ret |= writeReg(ES8311_SYSTEM_REG0D, 0x01);
794 ret |= writeReg(ES8311_ADC_REG15, 0x40);
795 ret |= writeReg(ES8311_DAC_REG37, 0x48);
796 ret |= writeReg(ES8311_GP_REG45, 0x00);
797
798 return ret;
799 }

802 error_t stop(codec_mode_t mode) {
803 error_t ret = RESULT_OK;
804 suspend();
805 return ret;
806 }
