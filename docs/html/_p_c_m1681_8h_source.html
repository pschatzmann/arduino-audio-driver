148 bool setVolume(uint8_t channel, uint8_t volume, bool apply = true) {
149 if (channel == 0 || channel > N_CHANNELS) return false;
150 if (volume > 100) volume = 100;
151
152 PCM1681AttenuationScale dams =
153 (reg_map[DAMS_REG] & 0x80) ? PCM1681AttenuationScale::Wide
154 : PCM1681AttenuationScale::Fine;
155
156 uint8_t attenuation =
157 (dams == PCM1681AttenuationScale::Wide)
158 ? (uint8_t)(101 * volume / 100 + 154 + 0.5)
159 : (uint8_t)(127 * volume / 100 + 128 + 0.5);
160
161 uint8_t reg = attenuationReg(channel);
162 reg_map[reg] = attenuation;
163
164 return apply ? writeReg(reg, attenuation) : true;
165 }

183 bool setMute(uint8_t channel, bool mute, bool apply = true) {
184 if (channel == 0 || channel > N_CHANNELS) return false;
185
186 uint8_t reg;
187 uint8_t pos;
188 if (channel < 7) {
189 reg = MUTx_REG;
190 pos = channel - 1;
191 } else {
192 reg = MUT_OR_REG;
193 pos = channel - 7;
194 }
195
196 updateBit(reg, pos, mute);
197 return apply ? writeReg(reg, reg_map[reg]) : true;
198 }

285 bool applyProperties() {
286 bool rc = true;
287 for (uint8_t reg = 0; reg < N_REGISTERS; reg++) {
288 if (!isWriteable(reg)) continue;
289 rc &= writeReg(reg, reg_map[reg]);
290 }
291 return rc;
292 }

300 return isAccessible(reg) && reg != REG_14;

303 static uint8_t attenuationReg(uint8_t channel) {
304 switch (channel) {
305 case 7:
306 return AT7x_REG;
307 case 8:
308 return AT8x_REG;
309 default:
310 return channel; // AT1x_REG .. AT6x_REG == 1..6
311 }

314 void updateBit(uint8_t reg, uint8_t pos, bool value) {
315 if (value) {
316 reg_map[reg] |= (uint8_t)(1U << pos);
317 } else {
318 reg_map[reg] &= (uint8_t)~(1U << pos);
319 }
320 }
